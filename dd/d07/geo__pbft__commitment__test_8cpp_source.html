46TEST(GeoCommitTest, GeoRequestFromOtherRegion) {
47 ResConfigData config_data = ResConfigData();
48 config_data.set_self_region_id(1);
49 std::vector<ReplicaInfo> replicas = {
50 GenerateReplicaInfo(1, "127.0.0.1", 1234),
51 GenerateReplicaInfo(2, "127.0.0.1", 1235),
52 GenerateReplicaInfo(3, "127.0.0.1", 1236),
53 GenerateReplicaInfo(4, "127.0.0.1", 1237)};
54 ResDBConfig config = ResDBConfig(
55 replicas, GenerateReplicaInfo(1, "127.0.0.1", 1234), config_data);
56 auto system_info = std::make_unique<SystemInfo>(config);
57 MockResDBReplicaClient replica_client;
58 std::unique_ptr<MockGeoGlobalExecutor> global_executor;
59 std::unique_ptr<GeoPBFTCommitment> commitment;
60
61 auto context = std::make_unique<Context>();
62 context->signature.set_signature("signature");
63
64 Request request;
65 request.set_seq(1);
66 request.mutable_region_info()->set_region_id(2);
67 BatchClientRequest client_request;
68 *client_request.mutable_committed_certs()->add_committed_certs() =
69 SignatureInfo();
70 client_request.SerializeToString(request.mutable_data());
71
72 global_executor = std::make_unique<MockGeoGlobalExecutor>(
73 std::make_unique<MockTransactionExecutorImpl>(), config);
74 std::promise<bool> done;
75 std::future<bool> done_future = done.get_future();
76 EXPECT_CALL(replica_client, BroadCast).Times(1);
77 EXPECT_CALL(*global_executor, Execute).Times(1).WillOnce(Invoke([&]() {
78 done.set_value(true);
79 return;
80 }));
81
82 MockSignatureVerifier verifier;
83 EXPECT_CALL(verifier, VerifyMessage(::testing::_, ::testing::_))
84 .WillOnce(Return(true));
85 commitment = std::make_unique<GeoPBFTCommitment>(
86 std::move(global_executor), config, std::move(system_info),
87 &replica_client, &verifier);
88 EXPECT_EQ(commitment->GeoProcessCcm(std::move(context),
89 std::make_unique<Request>(request)),
90 0);
91 done_future.get();
92}
