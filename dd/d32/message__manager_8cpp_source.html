34MessageManager::MessageManager(
35 const ResDBConfig& config,
36 std::unique_ptr<TransactionManager> transaction_manager,
37 CheckPointManager* checkpoint_manager, SystemInfo* system_info)
38 : config_(config),
39 queue_("executed"),
40 txn_db_(checkpoint_manager->GetTxnDB()),
41 system_info_(system_info),
42 checkpoint_manager_(checkpoint_manager),
43 transaction_executor_(std::make_unique<TransactionExecutor>(
44 config,
45 [&](std::unique_ptr<Request> request,
46 std::unique_ptr<BatchUserResponse> resp_msg) {
47 if (request->is_recovery()) {
48 return;
49 }
50 resp_msg->set_proxy_id(request->proxy_id());
51 resp_msg->set_seq(request->seq());
52 resp_msg->set_current_view(request->current_view());
53 resp_msg->set_primary_id(GetCurrentPrimary());
54 if (transaction_executor_->NeedResponse() &&
55 resp_msg->proxy_id() != 0) {
56 queue_.Push(std::move(resp_msg));
57 }
58 if (checkpoint_manager_) {
59 checkpoint_manager_->AddCommitData(std::move(request));
60 }
61 },
62 system_info_, std::move(transaction_manager))),
63 collector_pool_(std::make_unique<LockFreeCollectorPool>(
64 "txn", config_.GetMaxProcessTxn(), transaction_executor_.get(),
65 config_.GetConfigData().enable_viewchange())) {
66 global_stats_ = Stats::GetGlobalStats();
67 transaction_executor_->SetSeqUpdateNotifyFunc(
68 [&](uint64_t seq) { collector_pool_->Update(seq - 1); });
69 checkpoint_manager_->SetExecutor(transaction_executor_.get());
70}

98 std::unique_lock<std::mutex> lk(seq_mutex_);
99 uint32_t max_executed_seq = transaction_executor_->GetMaxPendingExecutedSeq();
100 global_stats_->SeqGap(next_seq_ - max_executed_seq);

114// Check if the request is valid.
115// 1. view is the same as the current view
116// 2. seq is larger or equal than the next execute seq.
117// 3. inside the water mark.
118bool MessageManager::IsValidMsg(const Request& request) {
119 if (request.type() == Request::TYPE_RESPONSE) {
120 return true;
121 }
122 // view should be the same as the current one.
123 if (static_cast<uint64_t>(request.current_view()) != GetCurrentView()) {
124 LOG(ERROR) << "message view :[" << request.current_view()
125 << "] is older than the cur view :[" << GetCurrentView() << "]";
126 return false;
127 }
128
129 if (static_cast<uint64_t>(request.seq()) <
130 transaction_executor_->GetMaxPendingExecutedSeq()) {
131 return false;
132 }
133
134 return true;
135}

137bool MessageManager::MayConsensusChangeStatus(
138 int type, int received_count, std::atomic<TransactionStatue>* status,
139 bool ret) {
140 switch (type) {
141 case Request::TYPE_PRE_PREPARE:
142 if (*status == TransactionStatue::None) {
143 TransactionStatue old_status = TransactionStatue::None;
144 return status->compare_exchange_strong(
145 old_status, TransactionStatue::READY_PREPARE,
146 std::memory_order_acq_rel, std::memory_order_acq_rel);
147 }
148 break;
149 case Request::TYPE_PREPARE:
150 if (*status == TransactionStatue::READY_PREPARE &&
151 config_.GetMinDataReceiveNum() <= received_count) {
152 TransactionStatue old_status = TransactionStatue::READY_PREPARE;
153 return status->compare_exchange_strong(
154 old_status, TransactionStatue::READY_COMMIT,
155 std::memory_order_acq_rel, std::memory_order_acq_rel);
156 }
157 break;
158 case Request::TYPE_COMMIT:
159 if (*status == TransactionStatue::READY_COMMIT &&
160 config_.GetMinDataReceiveNum() <= received_count) {
161 TransactionStatue old_status = TransactionStatue::READY_COMMIT;
162 return status->compare_exchange_strong(
163 old_status, TransactionStatue::READY_EXECUTE,
164 std::memory_order_acq_rel, std::memory_order_acq_rel);
165 return true;
166 }
167 break;
168 }
169 return ret;
170}

172// Add commit messages and return the number of messages have been received.
173// The commit messages only include post(pre-prepare), prepare and commit
174// messages. Messages are handled by state (PREPARE,COMMIT,READY_EXECUTE).
175
176// If there are enough messages and the state is changed after adding the
177// message, return 1, otherwise return 0. Return -2 if the request is not valid.
178CollectorResultCode MessageManager::AddConsensusMsg(
179 const SignatureInfo& signature, std::unique_ptr<Request> request) {
180 if (request == nullptr || !IsValidMsg(*request)) {
181 return CollectorResultCode::INVALID;
182 }
183 int type = request->type();
184 uint64_t seq = request->seq();
185 int resp_received_count = 0;
186 int proxy_id = request->proxy_id();
187
188 int ret = collector_pool_->GetCollector(seq)->AddRequest(
189 std::move(request), signature, type == Request::TYPE_PRE_PREPARE,
190 [&](const Request& request, int received_count,
191 TransactionCollector::CollectorDataType* data,
192 std::atomic<TransactionStatue>* status, bool force) {
193 if (MayConsensusChangeStatus(type, received_count, status, force)) {
194 resp_received_count = 1;
195 }
196 });
197 if (ret == 1) {
198 SetLastCommittedTime(proxy_id);
199 } else if (ret != 0) {
200 return CollectorResultCode::INVALID;
201 }
202 if (resp_received_count > 0) {
203 return CollectorResultCode::STATE_CHANGED;
204 }
205 return CollectorResultCode::OK;
206}

208RequestSet MessageManager::GetRequestSet(uint64_t min_seq, uint64_t max_seq) {
209 RequestSet ret;
210 std::unique_lock<std::mutex> lk(data_mutex_);
211 for (uint64_t i = min_seq; i <= max_seq; ++i) {
212 if (committed_data_.find(i) == committed_data_.end()) {
213 LOG(ERROR) << "seq :" << i << " doesn't exist";
214 continue;
215 }
216 RequestWithProof* request = ret.add_requests();
217 *request->mutable_request() = committed_data_[i];
218 request->set_seq(i);
219 for (const auto& request_info : committed_proof_[i]) {
220 RequestWithProof::RequestData* data = request->add_proofs();
221 *data->mutable_request() = *request_info->request;
222 *data->mutable_signature() = request_info->signature;
223 }
224 }
225 return ret;
226}
