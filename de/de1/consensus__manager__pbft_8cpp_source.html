144// The implementation of PBFT.
145int ConsensusManagerPBFT::ConsensusCommit(std::unique_ptr<Context> context,
146 std::unique_ptr<Request> request) {
147 // LOG(INFO) << "recv impl type:" << request->type() << " "
148 // << "sender id:" << request->sender_id();
149 // If it is in viewchange, push the request to the queue
150 // for the requests from the new view which come before
151 // the local new view done.
152 recovery_->AddRequest(context.get(), request.get());
153 if (config_.GetConfigData().enable_viewchange()) {
154 view_change_manager_->MayStart();
155 if (view_change_manager_->IsInViewChange()) {
156 switch (request->type()) {
157 case Request::TYPE_NEW_TXNS:
158 case Request::TYPE_PRE_PREPARE:
159 case Request::TYPE_PREPARE:
160 case Request::TYPE_COMMIT:
161 AddPendingRequest(std::move(context), std::move(request));
162 return 0;
163 }
164 } else {
165 while (true) {
166 auto new_request = PopPendingRequest();
167 if (!new_request.ok()) {
168 break;
169 }
170 InternalConsensusCommit(std::move((*new_request).first),
171 std::move((*new_request).second));
172 }
173 }
174 }
175 int ret = InternalConsensusCommit(std::move(context), std::move(request));
176 if (config_.GetConfigData().enable_viewchange()) {
177 if (ret == -4) {
178 while (true) {
179 auto new_request = PopComplainedRequest();
180 if (!new_request.ok()) {
181 break;
182 }
183 // LOG(ERROR) << "[POP COMPLAINED REQUEST]";
184 InternalConsensusCommit(std::move((*new_request).first),
185 std::move((*new_request).second));
186 }
187 }
188 }
189 return ret;
190}

192int ConsensusManagerPBFT::InternalConsensusCommit(
193 std::unique_ptr<Context> context, std::unique_ptr<Request> request) {
194 // LOG(INFO) << "recv impl type:" << request->type() << " "
195 // << "sender id:" << request->sender_id()<<" seq:"<<request->seq();
196
197 switch (request->type()) {
198 case Request::TYPE_CLIENT_REQUEST:
199 if (config_.IsPerformanceRunning()) {
200 return performance_manager_->StartEval();
201 }
202 return response_manager_->NewUserRequest(std::move(context),
203 std::move(request));
204 case Request::TYPE_RESPONSE:
205 if (config_.IsPerformanceRunning()) {
206 return performance_manager_->ProcessResponseMsg(std::move(context),
207 std::move(request));
208 }
209 return response_manager_->ProcessResponseMsg(std::move(context),
210 std::move(request));
211 case Request::TYPE_NEW_TXNS: {
212 uint64_t proxy_id = request->proxy_id();
213 std::string hash = request->hash();
214 int ret = commitment_->ProcessNewRequest(std::move(context),
215 std::move(request));
216 if (ret == -3) {
217 std::pair<std::unique_ptr<Context>, std::unique_ptr<Request>>
218 request_complained;
219 {
220 std::lock_guard<std::mutex> lk(commitment_->rc_mutex_);
221
222 request_complained =
223 std::move(commitment_->request_complained_.front());
224 commitment_->request_complained_.pop();
225 }
226 AddComplainedRequest(std::move(request_complained.first),
227 std::move(request_complained.second));
228 view_change_manager_->AddComplaintTimer(proxy_id, hash);
229 }
230 return ret;
231 }
232 case Request::TYPE_PRE_PREPARE:
233 return commitment_->ProcessProposeMsg(std::move(context),
234 std::move(request));
235 case Request::TYPE_PREPARE:
236 return commitment_->ProcessPrepareMsg(std::move(context),
237 std::move(request));
238 case Request::TYPE_COMMIT:
239 return commitment_->ProcessCommitMsg(std::move(context),
240 std::move(request));
241 case Request::TYPE_CHECKPOINT:
242 return checkpoint_manager_->ProcessCheckPoint(std::move(context),
243 std::move(request));
244 case Request::TYPE_VIEWCHANGE:
245 return view_change_manager_->ProcessViewChange(std::move(context),
246 std::move(request));
247 case Request::TYPE_NEWVIEW:
248 return view_change_manager_->ProcessNewView(std::move(context),
249 std::move(request));
250 case Request::TYPE_QUERY:
251 return query_->ProcessQuery(std::move(context), std::move(request));
252 case Request::TYPE_REPLICA_STATE:
253 return query_->ProcessGetReplicaState(std::move(context),
254 std::move(request));
255 case Request::TYPE_CUSTOM_QUERY:
256 return query_->ProcessCustomQuery(std::move(context), std::move(request));
257 }
258 return 0;
259}
