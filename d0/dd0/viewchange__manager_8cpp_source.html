168bool ViewChangeManager::IsValidViewChangeMsg(
169 const ViewChangeMessage& view_change_message) {
170 if (view_change_message.view_number() <= system_info_->GetCurrentView()) {
171 LOG(ERROR) << "View number is not greater than current view:"
172 << view_change_message.view_number() << "("
173 << system_info_->GetCurrentView() << ")";
174 return false;
175 }
176
177 if (!checkpoint_manager_->IsValidCheckpointProof(
178 view_change_message.stable_ckpt())) {
179 LOG(ERROR) << "stable checkpoint invalid";
180 return false;
181 }
182
183 uint64_t stable_seq = view_change_message.stable_ckpt().seq();
184
185 for (const auto& prepared_msg : view_change_message.prepared_msg()) {
186 if (prepared_msg.seq() <= stable_seq) {
187 continue;
188 }
189 // If there is less than 2f+1 proof, reject.
190 if (prepared_msg.proof_size() < config_.GetMinDataReceiveNum()) {
191 LOG(ERROR) << "proof[" << prepared_msg.proof_size()
192 << "] not enough:" << config_.GetMinDataReceiveNum();
193 return false;
194 }
195 for (const auto& proof : prepared_msg.proof()) {
196 if (proof.request().seq() != prepared_msg.seq()) {
197 LOG(ERROR) << "proof seq not match";
198 return false;
199 }
200 std::string data;
201 proof.request().SerializeToString(&data);
202 if (!verifier_->VerifyMessage(data, proof.signature())) {
203 LOG(ERROR) << "proof signature not valid";
204 return false;
205 }
206 }
207 }
208 return true;
209}

235std::vector<std::unique_ptr<Request>> ViewChangeManager::GetPrepareMsg(
236 const NewViewMessage& new_view_message, bool need_sign) {
237 std::map<uint64_t, Request> prepared_msg; // <sequence, digest>
238 for (const auto& msg : new_view_message.viewchange_messages()) {
239 for (const auto& msg : msg.prepared_msg()) {
240 uint64_t seq = msg.seq();
241 prepared_msg[seq] = msg.proof(0).request();
242 }
243 }
244
245 uint64_t min_s = std::numeric_limits<uint64_t>::max();
246 for (const auto& msg : new_view_message.viewchange_messages()) {
247 min_s = std::min(min_s, msg.stable_ckpt().seq());
248 }
249
250 uint64_t max_seq = 1;
251 if (prepared_msg.size() > 0) {
252 max_seq = (--prepared_msg.end())->first;
253 }
254
255 LOG(INFO) << "[GP] min_s: " << min_s << " max_seq: " << max_seq;
256
257 std::vector<std::unique_ptr<Request>> redo_request;
258 // Resent all the request with the current view number.
259 for (auto i = min_s + 1; i <= max_seq; ++i) {
260 if (prepared_msg.find(i) == prepared_msg.end()) {
261 // for sequence hole, create a new request with empty data and
262 // sign by the new primary.
263 std::unique_ptr<Request> user_request = resdb::NewRequest(
264 Request::TYPE_PRE_PREPARE, Request(), config_.GetSelfInfo().id());
265 user_request->set_seq(i);
266 user_request->set_current_view(new_view_message.view_number());
267 user_request->set_hash("null" + std::to_string(i));
268
269 if (verifier_ && need_sign) {
270 std::string data;
271 auto signature_or = verifier_->SignMessage(data);
272 if (!signature_or.ok()) {
273 LOG(ERROR) << "Sign message fail";
274 continue;
275 }
276 *user_request->mutable_data_signature() = *signature_or;
277 }
278 redo_request.push_back(std::move(user_request));
279 } else {
280 std::unique_ptr<Request> commit_request = resdb::NewRequest(
281 Request::TYPE_COMMIT, prepared_msg[i], config_.GetSelfInfo().id());
282 commit_request->set_seq(i);
283 commit_request->set_current_view(new_view_message.view_number());
284 redo_request.push_back(std::move(commit_request));
285 }
286 }
287
288 return redo_request;
289}

291int ViewChangeManager::ProcessNewView(std::unique_ptr<Context> context,
292 std::unique_ptr<Request> request) {
293 NewViewMessage new_view_message;
294 if (!new_view_message.ParseFromString(request->data())) {
295 LOG(ERROR) << "Parsing new_view_msg failed.";
296 return -2;
297 }
298 LOG(INFO) << "Received NEW-VIEW for view " << new_view_message.view_number();
299 // Check if new view is from next expected primary
300 if (new_view_message.view_number() !=
301 system_info_->GetCurrentView() + view_change_counter_) {
302 LOG(ERROR) << "View number " << new_view_message.view_number()
303 << " is not the same as expected: "
304 << system_info_->GetCurrentView() + 1;
305 return -2;
306 }
307
308 uint64_t min_s = std::numeric_limits<uint64_t>::max(), max_s = 0;
309 // Verify each view change message as the new primary does.
310 for (const auto& msg : new_view_message.viewchange_messages()) {
311 min_s = std::min(min_s, msg.stable_ckpt().seq());
312 max_s = std::max(max_s, msg.stable_ckpt().seq());
313 if (!IsValidViewChangeMsg(msg)) {
314 LOG(ERROR) << "view change message in the new-view message is invalid";
315 return -2;
316 }
317 }
318
319 // check the re-calculated request is the same as the one in the request.
320 auto request_list = GetPrepareMsg(new_view_message, false);
321 if (request_list.size() !=
322 static_cast<size_t>(new_view_message.request_size())) {
323 LOG(ERROR) << "redo request list size not match:" << request_list.size()
324 << "- " << new_view_message.request_size();
325 return -2;
326 }
327
328 std::set<uint64_t> seq_set;
329 // only check the data.
330 for (size_t i = 0; i < request_list.size(); ++i) {
331 if (request_list[i]->data() != new_view_message.request(i).data()) {
332 LOG(ERROR) << "data not match";
333 return -2;
334 }
335 seq_set.insert(request_list[i]->seq());
336 }
337
338 LOG(ERROR) << "min_s: " << min_s << " max_s: " << max_s;
339
340 // Check if all the sequences in the committed list exist.
341 for (uint64_t i = min_s + 1; i <= max_s; i++) {
342 if (seq_set.find(i) == seq_set.end()) {
343 LOG(ERROR) << "Committed msg :" << i << " does exist";
344 return -2;
345 }
346 }
347
348 uint64_t max_seq = seq_set.empty() ? max_s : *(--seq_set.end());
349
350 SetCurrentViewAndNewPrimary(new_view_message.view_number());
351 message_manager_->SetNextSeq(max_seq + 1);
352 LOG(INFO) << "SetNexSeq: " << max_seq + 1;
353
354 // All is fine.
355 for (size_t i = 0; i < request_list.size(); ++i) {
356 if (new_view_message.request(i).type() ==
357 static_cast<int>(Request::TYPE_PRE_PREPARE)) {
358 new_view_message.request(i);
359 auto non_proposed_hashes =
360 collector_pool_->GetCollector(new_view_message.request(i).seq())
361 ->GetAllStoredHash();
362 for (auto& hash : non_proposed_hashes) {
363 duplicate_manager_->EraseProposed(hash);
364 }
365 replica_communicator_->SendMessage(new_view_message.request(i),
366 config_.GetSelfInfo());
367 } else {
368 if (new_view_message.request(i).seq() >
369 checkpoint_manager_->GetHighestPreparedSeq()) {
370 checkpoint_manager_->SetHighestPreparedSeq(
371 new_view_message.request(i).seq());
372 }
373 replica_communicator_->BroadCast(new_view_message.request(i));
374 }
375 }
376
377 ChangeStatue(ViewChangeStatus::NONE);
378 return config_.GetSelfInfo().id() == system_info_->GetPrimaryId() ? -4 : 0;
379}

381int ViewChangeManager::ProcessViewChange(std::unique_ptr<Context> context,
382 std::unique_ptr<Request> request) {
383 ViewChangeMessage viewchange_message;
384 if (!viewchange_message.ParseFromString(request->data())) {
385 LOG(ERROR) << "pase view change data fail";
386 return -2;
387 }
388
389 if (!IsValidViewChangeMsg(viewchange_message)) {
390 LOG(ERROR) << "view change msg not valid from:" << request->sender_id();
391 return -2;
392 }
393
394 LOG(ERROR) << "ViewChange message from " << request->sender_id();
395
396 size_t request_size = AddRequest(viewchange_message, request->sender_id());
397 if (request_size >= config_.GetMinDataReceiveNum()) {
398 // process new view
399 if (IsNextPrimary(viewchange_message.view_number())) {
400 std::lock_guard<std::mutex> lk(mutex_);
401 SendNewViewMsg(viewchange_message.view_number());
402 } else {
403 auto newview_timer = std::make_shared<ViewChangeTimeout>(
404 ViewChangeTimerType::TYPE_NEWVIEW, system_info_->GetCurrentView(),
405 config_.GetSelfInfo().id(), "null", GetCurrentTime(),
406 timeout_length_);
407 std::lock_guard<std::mutex> lk(vc_mutex_);
408 if (viewchange_timeout_min_heap_[config_.GetSelfInfo().id()].size() <
409 config_.GetMaxClientComplaintNum()) {
410 viewchange_timeout_min_heap_[config_.GetSelfInfo().id()].push(
411 newview_timer);
412 sem_post(&viewchange_timer_signal_);
413 }
414 }
415 ChangeStatue(ViewChangeStatus::READY_NEW_VIEW);
416 }
417 return 0;
418}

420void ViewChangeManager::SendNewViewMsg(uint64_t view_number) {
421 if (new_view_is_sent_) {
422 return;
423 }
424 new_view_is_sent_ = true;
425
426 // PBFT Paper - Primary determines the sequence number min-s of the latest
427 // stable checkpoint in V and the highest sequence number max-s in a prepare
428 // message in V
429 // uint64_t min_s = std::numeric_limits<uint64_t>::max();
430
431 std::lock_guard<std::mutex> lk(vc_mutex_);
432 auto requests = viewchange_request_[view_number];
433
434 NewViewMessage new_view_message;
435 new_view_message.set_view_number(view_number);
436
437 std::map<uint64_t, std::string> new_view_request; // <sequence, digest>
438 for (auto& it : requests) {
439 ViewChangeMessage& msg = it.second;
440 LOG(ERROR) << "msg.view_number(): " << msg.view_number()
441 << " view_number: " << view_number << " sender: " << it.first
442 << " msg.stable_ckpt(): " << msg.stable_ckpt().seq();
443 msg.set_view_number(view_number);
444 *new_view_message.add_viewchange_messages() = msg;
445 }
446
447 // Get the redo message from the primary. This could not be done
448 // in each replica because the primary has to signed some empty
449 // request if needed.
450
451 std::vector<std::unique_ptr<Request>> request_list =
452 GetPrepareMsg(new_view_message);
453 for (const auto& request : request_list) {
454 *new_view_message.add_request() = *request;
455 }
456
457 // Broadcast my new view request.
458 std::unique_ptr<Request> request =
459 NewRequest(Request::TYPE_NEWVIEW, Request(), config_.GetSelfInfo().id());
460 new_view_message.SerializeToString(request->mutable_data());
461 replica_communicator_->BroadCast(*request);
462}

464void ViewChangeManager::SendViewChangeMsg() {
465 // PBFT Paper - <VIEW-CHANGE, v + x, n, C, P)
466 ViewChangeMessage view_change_message;
467 // v + x (view number of the next expected primary)
468 view_change_message.set_view_number(system_info_->GetCurrentView() +
469 view_change_counter_);
470
471 LOG(ERROR) << "current view: " << system_info_->GetCurrentView()
472 << " view number: " << view_change_message.view_number()
473 << " view_change_counter_ " << view_change_counter_;
474 assert(view_change_message.view_number() ==
475 system_info_->GetCurrentView() + view_change_counter_);
476
477 // n (sequence number of the latest checkpoint) and C (proof for the stable
478 // checkpoint)
479 *view_change_message.mutable_stable_ckpt() =
480 checkpoint_manager_->GetStableCheckpointWithVotes();
481
482 // P - P is a set containing a set Pm for each request m that prepared at i
483 // with a sequence number higher than n.
484 int max_seq = checkpoint_manager_->GetHighestPreparedSeq();
485 LOG(INFO) << "Check prepared or committed txns from "
486 << view_change_message.stable_ckpt().seq() + 1 << " to " << max_seq;
487
488 for (int i = view_change_message.stable_ckpt().seq() + 1; i <= max_seq; ++i) {
489 // seq i has been prepared or committed.
490 if (message_manager_->GetTransactionState(i) >=
491 TransactionStatue::READY_COMMIT) {
492 std::vector<RequestInfo> proof_info =
493 message_manager_->GetPreparedProof(i);
494 assert(proof_info.size() >= config_.GetMinDataReceiveNum());
495 auto txn = view_change_message.add_prepared_msg();
496 txn->set_seq(i);
497 for (const auto& info : proof_info) {
498 auto proof = txn->add_proof();
499 *proof->mutable_request() = *info.request;
500 *proof->mutable_signature() = info.signature;
501 }
502 }
503 }
504
505 // Broadcast my view change request.
506 std::unique_ptr<Request> request = NewRequest(
507 Request::TYPE_VIEWCHANGE, Request(), config_.GetSelfInfo().id());
508 view_change_message.SerializeToString(request->mutable_data());
509 replica_communicator_->BroadCast(*request);
510}

528void ViewChangeManager::MonitoringViewChangeTimeOut() {
529 while (!stop_) {
530 // [DK3] After timer is out, the client will check if the corresponding
531 // client request has recevied sufficient valid responses
532 sem_wait(&viewchange_timer_signal_);
533 vc_mutex_.lock();
534 bool empty = true;
535 std::shared_ptr<ViewChangeTimeout> viewchange_timeout;
536 for (auto& heap : viewchange_timeout_min_heap_) {
537 if (!heap.second.empty()) {
538 viewchange_timeout = heap.second.top();
539 heap.second.pop();
540 vc_mutex_.unlock();
541 empty = false;
542 break;
543 }
544 }
545 if (empty) {
546 vc_mutex_.unlock();
547 continue;
548 }
549 auto current_time = GetCurrentTime();
550 if (viewchange_timeout->timeout_time > current_time) {
551 usleep(viewchange_timeout->timeout_time - current_time);
552 }
553 // [DK3] if not enough responses are received, the client broadcasts the
554 // client request to all replicas
555 if (viewchange_timeout->type == ViewChangeTimerType::TYPE_NEWVIEW) {
556 if (status_ == ViewChangeStatus::READY_NEW_VIEW &&
557 viewchange_timeout->view == system_info_->GetCurrentView()) {
558 // [DK12] if the replicas cannot receive a newview message in a timely
559 // manner, they will enter the next view and starts a new round of
560 // viewchange. SetCurrentViewAndNewPrimary(viewchange_timeout->view +
561 // 1);
562 LOG(ERROR) << "It is time to start a new viewchange";
563 checkpoint_manager_->TimeoutHandler();
564 }
565 } else if (viewchange_timeout->type ==
566 ViewChangeTimerType::TYPE_VIEWCHANGE) {
567 // [DK9] if the primary cannot get enough viewchange messages before the
568 // timer is out, then it broadcasts its viewchanges messages and starts
569 // the timer again.
570 if (status_ == ViewChangeStatus::READY_VIEW_CHANGE &&
571 viewchange_timeout->view == system_info_->GetCurrentView()) {
572 LOG(ERROR) << "It is time to rebroacast viewchange messages";
573 ChangeStatue(ViewChangeStatus::VIEW_CHANGE_FAIL);
574 checkpoint_manager_->TimeoutHandler();
575 }
576 } else if (viewchange_timeout->type ==
577 ViewChangeTimerType::TYPE_COMPLAINT) {
578 // [DK7] if the primary does not broadcast the request in a timely manner,
579 // the replica starts a viewchange
580 if (complaining_clients_[viewchange_timeout->proxy_id]
581 .CountViewChangeTimeout(viewchange_timeout->hash)) {
582 complaining_clients_[viewchange_timeout->proxy_id]
583 .EraseViewChangeTimeout(viewchange_timeout->hash);
584 }
585 std::lock_guard<std::mutex> lk(status_mutex_);
586 if (status_ == ViewChangeStatus::NONE &&
587 viewchange_timeout->view == system_info_->GetCurrentView()) {
588 if (viewchange_timeout->start_time >=
589 message_manager_->GetLastCommittedTime(
590 viewchange_timeout->proxy_id)) {
591 LOG(ERROR) << "It is time to start a viewchange";
592 checkpoint_manager_->TimeoutHandler();
593 assert(status_ == ViewChangeStatus::READY_VIEW_CHANGE);
594 }
595 }
596 }
597 }

600void ViewChangeManager::MonitoringCheckpointState() {
601 uint64_t last_seq_value = 0;
602 while (!stop_) {
603 sem_wait(checkpoint_manager_->CommitableSeqSignal());
604 auto value = checkpoint_manager_->GetCommittableSeq();
605 if (last_seq_value != value) {
606 last_seq_value = value;
607 if (IsInViewChange()) {
608 ChangeStatue(ViewChangeStatus::NONE);
609 }
610 }
