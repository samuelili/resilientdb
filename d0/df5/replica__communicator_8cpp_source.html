65ReplicaCommunicator::~ReplicaCommunicator() {
66 is_running_ = false;
67 if (broadcast_thread_.joinable()) {
68 broadcast_thread_.join();
69 }
70 if (is_use_long_conn_) {
71 for (auto& cli : client_pools_) {
72 cli.second.reset();
73 }
74 worker_.reset();
75 worker_ = nullptr;
76 io_service_.stop();
77 for (auto& worker_th : worker_threads_) {
78 if (worker_th.joinable()) {
79 worker_th.join();

117void ReplicaCommunicator::StartBroadcastInBackGround() {
118 is_running_ = true;
119 broadcast_thread_ = std::thread([&]() {
120 while (IsRunning()) {
121 std::vector<std::unique_ptr<QueueItem>> batch_req =
122 batch_queue_.Pop(10000);
123 if (batch_req.empty()) {
124 continue;
125 }
126 BroadcastData broadcast_data;
127 for (auto& queue_item : batch_req) {
128 broadcast_data.add_data()->swap(queue_item->data);
129 }
130
131 global_stats_->SendBroadCastMsg(broadcast_data.data_size());
132 int ret = SendMessageFromPool(broadcast_data, replicas_);
133 if (ret < 0) {
134 LOG(ERROR) << "broadcast request fail:";
135 }
136 }
137 });
138}

141 global_stats_->BroadCastMsg();

165int ReplicaCommunicator::SendBatchMessage(
166 const std::vector<std::unique_ptr<Request>>& messages,
167 const ReplicaInfo& replica_info) {
168 if (is_use_long_conn_) {
169 BroadcastData broadcast_data;
170 for (const auto& message : messages) {
171 std::string data = NetChannel::GetRawMessageString(*message, verifier_);
172 broadcast_data.add_data()->swap(data);
173 }
174 return SendMessageFromPool(broadcast_data, {replica_info});
175 } else {
176 int ret = 0;
177 for (const auto& message : messages) {
178 ret += SendMessageInternal(*message, {replica_info});

184int ReplicaCommunicator::SendMessageFromPool(
185 const google::protobuf::Message& message,
186 const std::vector<ReplicaInfo>& replicas) {
187 int ret = 0;
188 std::string data;
189 message.SerializeToString(&data);
190 global_stats_->SendBroadCastMsgPerRep();
191 std::lock_guard<std::mutex> lk(mutex_);
192 for (const auto& replica : replicas) {
193 auto client = GetClientFromPool(replica.ip(), replica.port());
194 if (client == nullptr) {
195 continue;
196 }
197 if (client->SendMessage(data) == 0) {
198 ret++;
199 } else {
200 LOG(ERROR) << "send to:" << replica.ip() << " fail";
201 }
202 }
203 return ret;
204}

206int ReplicaCommunicator::SendMessageInternal(
207 const google::protobuf::Message& message,
208 const std::vector<ReplicaInfo>& replicas) {
209 int ret = 0;
210 for (const auto& replica : replicas) {
211 auto client = GetClient(replica.ip(), replica.port());
212 if (client == nullptr) {
213 continue;
214 }
215 if (verifier_ != nullptr) {
216 client->SetSignatureVerifier(verifier_);
217 }
218 if (client->SendRawMessage(message) == 0) {
219 ret++;
220 }
221 }
222 return ret;

225AsyncReplicaClient* ReplicaCommunicator::GetClientFromPool(
226 const std::string& ip, int port) {
227 if (client_pools_.find(std::make_pair(ip, port)) == client_pools_.end()) {
228 auto client = std::make_unique<AsyncReplicaClient>(
229 &io_service_, ip, port + (is_use_long_conn_ ? 10000 : 0), true);
230 client_pools_[std::make_pair(ip, port)] = std::move(client);
231 }
232 return client_pools_[std::make_pair(ip, port)].get();
233}

247void ReplicaCommunicator::SendMessage(const google::protobuf::Message& message,
248 int64_t node_id) {
249 ReplicaInfo target_replica;
250 for (const auto& replica : replicas_) {
251 if (replica.id() == node_id) {
252 target_replica = replica;
253 break;
254 }
255 }
256 if (target_replica.ip().empty()) {
257 for (const auto& replica : GetClientReplicas()) {
258 if (replica.id() == node_id) {
259 target_replica = replica;
260 break;
261 }
262 }
263 }
264
265 if (target_replica.ip().empty()) {
266 LOG(ERROR) << "no replica info";
267 return;
268 }
269
270 int ret = SendMessage(message, target_replica);
271 if (ret < 0) {
272 LOG(ERROR) << "broadcast request fail:";
273 }
274}
