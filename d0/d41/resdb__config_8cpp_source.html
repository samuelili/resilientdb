48ResDBConfig::ResDBConfig(const ResConfigData& config_data,
49 const ReplicaInfo& self_info,
50 const KeyInfo& private_key,
51 const CertificateInfo& public_key_cert_info)
52 : config_data_(config_data),
53 self_info_(self_info),
54 private_key_(private_key),
55 public_key_cert_info_(public_key_cert_info) {
56 for (const auto& region : config_data.region()) {
57 if (region.region_id() == config_data.self_region_id()) {
58 LOG(INFO) << "get region info:" << region.DebugString();
59 for (const auto& replica : region.replica_info()) {
60 replicas_.push_back(replica);
61 }
62 LOG(INFO) << "get region config server size:"
63 << region.replica_info_size();
64 break;
65 }
66 }
67 if (config_data_.view_change_timeout_ms() == 0) {
68 config_data_.set_view_change_timeout_ms(viewchange_commit_timeout_ms_);
69 }
70 if (config_data_.client_batch_num() == 0) {
71 config_data_.set_client_batch_num(client_batch_num_);
72 }
73 if (config_data_.worker_num() == 0) {
74 config_data_.set_worker_num(worker_num_);
75 }
76 if (config_data_.input_worker_num() == 0) {
77 config_data_.set_input_worker_num(input_worker_num_);
78 }
79 if (config_data_.output_worker_num() == 0) {
80 config_data_.set_output_worker_num(output_worker_num_);
81 }
82 if (config_data_.tcp_batch_num() == 0) {
83 config_data_.set_tcp_batch_num(100);
84 }
85}

87void ResDBConfig::SetConfigData(const ResConfigData& config_data) {
88 config_data_ = config_data;
89 replicas_.clear();
90 for (const auto& region : config_data.region()) {
91 if (region.region_id() == config_data.self_region_id()) {
92 LOG(INFO) << "get region info:" << region.DebugString();
93 for (const auto& replica : region.replica_info()) {
94 replicas_.push_back(replica);
95 }
96 LOG(INFO) << "get region config server size:"
97 << region.replica_info_size();
98 break;
99 }
100 }
101 if (config_data_.view_change_timeout_ms() == 0) {
102 config_data_.set_view_change_timeout_ms(viewchange_commit_timeout_ms_);
103 }
104}
