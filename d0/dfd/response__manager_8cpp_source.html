97// =================== response ========================
98// handle the response message. If receive f+1 commit messages, send back to the
99// caller.
100int ResponseManager::ProcessResponseMsg(std::unique_ptr<Context> context,
101 std::unique_ptr<Request> request) {
102 std::unique_ptr<Request> response;
103 // Add the response message, and use the call back to collect the received
104 // messages.
105 // The callback will be triggered if it received f+1 messages.
106 if (request->ret() == -2) {
107 LOG(ERROR) << "get response fail:" << request->ret();
108 send_num_--;
109 return 0;
110 }
111 CollectorResultCode ret =
112 AddResponseMsg(context->signature, std::move(request),
113 [&](const Request& request,
114 const TransactionCollector::CollectorDataType*) {
115 response = std::make_unique<Request>(request);
116 return;
117 });
118
119 if (ret == CollectorResultCode::STATE_CHANGED) {
120 BatchUserResponse batch_response;
121 if (batch_response.ParseFromString(response->data())) {
122 SendResponseToClient(batch_response);
123 } else {
124 LOG(ERROR) << "parse response fail:";
125 }
126 }
127 return ret == CollectorResultCode::INVALID ? -2 : 0;
128}

130bool ResponseManager::MayConsensusChangeStatus(
131 int type, int received_count, std::atomic<TransactionStatue>* status) {
132 switch (type) {
133 case Request::TYPE_RESPONSE:
134 // if receive f+1 response results, ack to the caller.
135 if (*status == TransactionStatue::None &&
136 config_.GetMinClientReceiveNum() <= received_count) {
137 TransactionStatue old_status = TransactionStatue::None;
138 return status->compare_exchange_strong(
139 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
140 std::memory_order_acq_rel);
141 }
142 break;
143 }
144 return false;
145}

147CollectorResultCode ResponseManager::AddResponseMsg(
148 const SignatureInfo& signature, std::unique_ptr<Request> request,
149 std::function<void(const Request&,
150 const TransactionCollector::CollectorDataType*)>
151 response_call_back) {
152 if (request == nullptr) {
153 return CollectorResultCode::INVALID;
154 }
155
156 int type = request->type();
157 uint64_t seq = request->seq();
158 int resp_received_count = 0;
159 int ret = collector_pool_->GetCollector(seq)->AddRequest(
160 std::move(request), signature, false,
161 [&](const Request& request, int received_count,
162 TransactionCollector::CollectorDataType* data,
163 std::atomic<TransactionStatue>* status, bool force) {
164 if (MayConsensusChangeStatus(type, received_count, status)) {
165 resp_received_count = 1;
166 response_call_back(request, data);
167 }
168 });
169 if (ret != 0) {
170 return CollectorResultCode::INVALID;
171 }
172 if (resp_received_count > 0) {
173 collector_pool_->Update(seq);
174 return CollectorResultCode::STATE_CHANGED;
175 }
176 return CollectorResultCode::OK;
177}

179void ResponseManager::SendResponseToClient(
180 const BatchUserResponse& batch_response) {
181 uint64_t create_time = batch_response.createtime();
182 uint64_t local_id = batch_response.local_id();
183 if (create_time > 0) {
184 uint64_t run_time = GetCurrentTime() - create_time;
185 global_stats_->AddLatency(run_time);
186 } else {
187 LOG(ERROR) << "seq:" << local_id << " no resp";
188 }
189 send_num_--;
190
191 if (config_.IsPerformanceRunning()) {
192 return;
193 }
194
195 std::vector<std::unique_ptr<Context>> context_list =
196 FetchContextList(batch_response.local_id());
197 if (context_list.empty()) {
198 LOG(ERROR) << "context list is empty. local id:"
199 << batch_response.local_id();
200 return;
201 }
202
203 for (size_t i = 0; i < context_list.size(); ++i) {
204 auto& context = context_list[i];
205 if (context->client == nullptr) {
206 LOG(ERROR) << " no channel:";
207 continue;
208 }
209 int ret = context->client->SendRawMessageData(batch_response.response(i));
210 if (ret) {
211 LOG(ERROR) << "send resp fail ret:" << ret;
212 }
213 }
214}

216// =================== request ========================
217int ResponseManager::BatchProposeMsg() {
218 LOG(INFO) << "batch wait time:" << config_.ClientBatchWaitTimeMS()
219 << " batch num:" << config_.ClientBatchNum();
220 std::vector<std::unique_ptr<QueueItem>> batch_req;
221 while (!stop_) {
222 if (send_num_ > config_.GetMaxProcessTxn()) {
223 LOG(ERROR) << "send num too high, wait:" << send_num_;
224 usleep(100);
225 continue;
226 }
227 if (batch_req.size() < config_.ClientBatchNum()) {
228 std::unique_ptr<QueueItem> item =
229 batch_queue_.Pop(config_.ClientBatchWaitTimeMS());
230 if (item != nullptr) {
231 batch_req.push_back(std::move(item));
232 if (batch_req.size() < config_.ClientBatchNum()) {
233 continue;
234 }
235 }
236 }
237 if (batch_req.empty()) {
238 continue;
239 }
240 int ret = DoBatch(batch_req);
241 batch_req.clear();
242 if (ret != 0) {
243 Response response;
244 response.set_result(Response::ERROR);
245 for (size_t i = 0; i < batch_req.size(); ++i) {
246 if (batch_req[i]->context && batch_req[i]->context->client) {
247 int ret = batch_req[i]->context->client->SendRawMessage(response);
248 if (ret) {
249 LOG(ERROR) << "send resp" << response.DebugString()
250 << " fail ret:" << ret;
251 }
252 }
253 }
254 }
255 }
256 return 0;
257}

259int ResponseManager::DoBatch(
260 const std::vector<std::unique_ptr<QueueItem>>& batch_req) {
261 auto new_request =
262 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
263 if (new_request == nullptr) {
264 return -2;
265 }
266 std::vector<std::unique_ptr<Context>> context_list;
267
268 BatchUserRequest batch_request;
269 for (size_t i = 0; i < batch_req.size(); ++i) {
270 BatchUserRequest::UserRequest* req = batch_request.add_user_requests();
271 *req->mutable_request() = *batch_req[i]->user_request.get();
272 *req->mutable_signature() = batch_req[i]->context->signature;
273 req->set_id(i);
274 context_list.push_back(std::move(batch_req[i]->context));
275 }
276
277 if (!config_.IsPerformanceRunning()) {
278 LOG(ERROR) << "add context list:" << new_request->seq()
279 << " list size:" << context_list.size();
280 batch_request.set_local_id(local_id_);
281 int ret = AddContextList(std::move(context_list), local_id_++);
282 if (ret != 0) {
283 LOG(ERROR) << "add context list fail:";
284 return ret;
285 }
286 }
287 batch_request.set_createtime(GetCurrentTime());
288 std::string data;
289 batch_request.SerializeToString(&data);
290 if (verifier_) {
291 auto signature_or = verifier_->SignMessage(data);
292 if (!signature_or.ok()) {
293 LOG(ERROR) << "Sign message fail";
294 return -2;
295 }
296 *new_request->mutable_data_signature() = *signature_or;
297 }
298
299 batch_request.SerializeToString(new_request->mutable_data());
300 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
301 new_request->set_proxy_id(config_.GetSelfInfo().id());
302 replica_communicator_->SendMessage(*new_request, GetPrimary());
303 send_num_++;
304 LOG(INFO) << "send msg to primary:" << GetPrimary()
305 << " batch size:" << batch_req.size();
306 return 0;
307}
