59void AsyncAcceptor::Session::StartRead() {
60 if (status_ == 0) {
61 // read len
62 recv_buffer_ = reinterpret_cast<char*>(&data_size_);
63 data_size_ = 0;
64 need_size_ = sizeof(data_size_);
65 current_idx_ = 0;
66 memset(recv_buffer_, 0, need_size_);
67 OnRead();
68 } else {
69 need_size_ = data_size_;
70 current_idx_ = 0;
71 if (recv_buffer_) {
72 delete recv_buffer_;
73 }
74 recv_buffer_ = new char[need_size_];
75 memset(recv_buffer_, 0, need_size_);
76 OnRead();
77 }
78}

80void AsyncAcceptor::Session::ReadDone() {
81 if (status_ == 1) {
82 call_back_func_(recv_buffer_, data_size_);
83 delete recv_buffer_;
84 } else {
85 data_size_ = *reinterpret_cast<size_t*>(recv_buffer_);
86 if (data_size_ > 1e6) {
87 LOG(ERROR) << "read data size:" << data_size_
88 << " data size:" << sizeof(data_size_) << " close socket";
89 Close();
90 return;
91 }
92 }
93 status_ ^= 1;
94 recv_buffer_ = nullptr;
95 // continue to read next msg.
96 StartRead();
97}

99void AsyncAcceptor::Session::OnRead() {
100 client_socket_.async_read_some(
101 boost::asio::buffer(recv_buffer_ + current_idx_,
102 need_size_ - current_idx_),
103 [&](const boost::system::error_code& error, // Result of operation.
104 std::size_t bytes_transferred) {
105 if (error || bytes_transferred == 0) {
106 Close();
107 } else {
108 current_idx_ += bytes_transferred;
109 if (current_idx_ >= need_size_) {
110 ReadDone();
111 } else {
112 OnRead();
113 }
114 }
115 });
116}

118AsyncAcceptor::AsyncAcceptor(const std::string& ip, int port, int thread_num,
119 CallBack call_back_func)
120 : endpoint_(boost::asio::ip::address::from_string(ip), port),
121 acceptor_(io_service_, endpoint_),
122 call_back_func_(call_back_func) {
123 worker_ = std::make_unique<boost::asio::io_service::work>(io_service_);
124 for (int i = 0; i < thread_num; ++i) {
125 worker_thread_.push_back(std::thread([&]() { io_service_.run(); }));
126 }
127}

129AsyncAcceptor::~AsyncAcceptor() {
130 worker_.reset();
131 worker_ = nullptr;
132 io_service_.stop();
133
134 for (auto& sess : sessions_) {
135 if (sess) {
136 sess->Close();
137 }
138 }
139
140 for (auto& worker : worker_thread_) {
141 if (worker.joinable()) {
142 worker.join();
143 }
144 }
145}

147void AsyncAcceptor::StartAccept() {
148 boost::shared_ptr<Session> client_session(
149 new Session(&io_service_, call_back_func_));
150 acceptor_.async_accept(*client_session->GetSocket(),
151 std::bind(&AsyncAcceptor::OnAccept, this,
152 client_session, std::placeholders::_1));
153}

155void AsyncAcceptor::OnAccept(boost::shared_ptr<Session> client_session,
156 const boost::system::error_code ec) {
157 if (ec) {
158 LOG(ERROR) << " accept fail";
159 return;
160 }
161
162 StartAccept(); // Add the next accept event.
163
164 sessions_.push_back(client_session);
165 client_session->StartRead();
166}
