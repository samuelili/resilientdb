35void AppendArgToInput(std::vector<uint8_t>& code, const uint256_t& arg) {
36 const auto pre_size = code.size();
37 code.resize(pre_size + 32u);
38 eevm::to_big_endian(arg, code.data() + pre_size);

59Address ContractManager::DeployContract(const Address& owner_address,
60 const DeployInfo& deploy_info) {
61 const auto contract_address =
62 AddressManager::CreateContractAddress(owner_address);
63
64 auto contract_constructor = eevm::to_bytes(deploy_info.contract_bin());
65 for (const std::string& param : deploy_info.init_param()) {
66 AppendArgToInput(contract_constructor, param);
67 }
68
69 try {
70 auto contract = gs_->create(contract_address, 0u, contract_constructor);
71
72 auto result = Execute(owner_address, contract_address, {});
73 if (result.ok()) {
74 // set the initialized class context code.
75 contract.acc.set_code(std::move(*result));
76
77 for (const auto& info : deploy_info.func_info()) {
78 SetFuncAddress(contract_address, info);
79 }
80 return contract.acc.get_address();
81 } else {
82 gs_->remove(contract_address);
83 return 0;
84 }
85 } catch (...) {
86 LOG(ERROR) << "Deploy throw expection";
87 return 0;
88 }
89}

100absl::StatusOr<std::string> ContractManager::ExecContract(
101 const Address& caller_address, const Address& contract_address,
102 const Params& func_param) {
103 std::string func_addr =
104 GetFuncAddress(contract_address, func_param.func_name());
105 if (func_addr.empty()) {
106 LOG(ERROR) << "no fouction:" << func_param.func_name();
107 return absl::InvalidArgumentError("Func not exist.");
108 }
109
110 std::vector<uint8_t> inputs = eevm::to_bytes(func_addr);
111 for (const std::string& param : func_param.param()) {
112 AppendArgToInput(inputs, param);
113 }
114
115 auto result = Execute(caller_address, contract_address, inputs);
116
117 if (result.ok()) {
118 return eevm::to_hex_string(*result);
119 }
120 return result.status();
121}

123absl::StatusOr<std::vector<uint8_t>> ContractManager::Execute(
124 const Address& caller_address, const Address& contract_address,
125 const std::vector<uint8_t>& input) {
126 // Ignore any logs produced by this transaction
127 eevm::NullLogHandler ignore;
128 eevm::Transaction tx(caller_address, ignore);
129
130 // Record a trace to aid debugging
131 eevm::Trace tr;
132 eevm::Processor p(*gs_);
133
134 // Run the transaction
135 try {
136 const auto exec_result =
137 p.run(tx, caller_address, gs_->get(contract_address), input, 0u, &tr);
138
139 if (exec_result.er != eevm::ExitReason::returned) {
140 // Print the trace if nothing was returned
141 if (exec_result.er == eevm::ExitReason::threw) {
142 return absl::InternalError(
143 fmt::format("Execution error: {}", exec_result.exmsg));
144 }
145 return absl::InternalError("Deployment did not return");
146 }
147 return exec_result.output;
148 } catch (...) {
149 return absl::InternalError(fmt::format("Execution error:"));
150 }
151}
