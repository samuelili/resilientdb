92// Keep Boardcast the public keys to others.
93void ConsensusManager::HeartBeat() {
94 LOG(INFO) << "heart beat start";
95 int sleep_time = 1;
96 std::mutex mutex;
97 std::condition_variable cv;
98 while (IsRunning()) {
99 auto keys = verifier_->GetAllPublicKeys();
100
101 std::vector<ReplicaInfo> replicas = GetAllReplicas();
102 LOG(ERROR) << "all replicas:" << replicas.size();
103 std::vector<ReplicaInfo> client_replicas = GetClientReplicas();
104 HeartBeatInfo hb_info;
105 for (const auto& key : keys) {
106 *hb_info.add_public_keys() = key;
107 }
108 for (const auto& client : client_replicas) {
109 replicas.push_back(client);
110 }
111 auto client = GetReplicaClient(replicas, false);
112 if (client == nullptr) {
113 continue;
114 }
115
116 // If it is not a client node, broadcost the current primary to the client.
117 if (config_.GetPublicKeyCertificateInfo()
118 .public_key()
119 .public_key_info()
120 .type() == CertificateKeyInfo::REPLICA) {
121 hb_info.set_primary(GetPrimary());
122 hb_info.set_version(GetVersion());
123 }
124 LOG(ERROR) << " server:" << config_.GetSelfInfo().id() << " sends HB"
125 << " is ready:" << is_ready_
126 << " client size:" << client_replicas.size()
127 << " svr size:" << replicas.size();
128
129 Request request;
130 request.set_type(Request::TYPE_HEART_BEAT);
131 request.mutable_region_info()->set_region_id(
132 config_.GetConfigData().self_region_id());
133 hb_info.SerializeToString(request.mutable_data());
134
135 int ret = client->SendHeartBeat(request);
136 if (ret <= 0) {
137 LOG(ERROR) << " server:" << config_.GetSelfInfo().id()
138 << " sends HB fail:" << ret;
139 }
140 std::unique_lock<std::mutex> lk(mutex);
141 cv.wait_for(lk, std::chrono::microseconds(sleep_time * 1000000),
142 [&] { return !IsRunning(); });
143 if (is_ready_) {
144 if (config_.IsTestMode()) {
145 sleep_time = 1;
146 } else {
147 sleep_time = 60 * 2;
148 }
149 }
150 }
151}

153// Porcess the packages received from the network.
154// context contains the client socket which can be used for sending response to
155// the client, the signature for the request will be filled inside the context
156// when parsing the message.
157int ConsensusManager::Process(std::unique_ptr<Context> context,
158 std::unique_ptr<DataInfo> request_info) {
159 global_stats_->IncClientCall();
160 // Decode the whole message, it includes the certificate and data.
161 ResDBMessage message;
162 if (!message.ParseFromArray(request_info->buff, request_info->data_len)) {
163 LOG(ERROR) << "parse data info fail";
164 return -1;
165 }
166
167 // Check if the certificate is valid.
168 if (message.has_signature() && verifier_) {
169 bool valid = verifier_->VerifyMessage(message.data(), message.signature());
170 if (!valid) {
171 LOG(ERROR) << "request is not valid:"
172 << message.signature().DebugString();
173 LOG(ERROR) << " msg:" << message.data().size();
174 return -2;
175 }
176 } else {
177 }
178
179 std::unique_ptr<Request> request = std::make_unique<Request>();
180 if (!request->ParseFromString(message.data())) {
181 LOG(ERROR) << "parse data info fail";
182 return -1;
183 }
184
185 std::string tmp;
186 if (!request->SerializeToString(&tmp)) {
187 return -1;
188 }
189
190 // forward the signature to the request so that it can be included in the
191 // request/response set if needed.
192 context->signature = message.signature();
193 // LOG(ERROR) << "======= server:" << config_.GetSelfInfo().id()
194 // << " get request type:" << request->type()
195 // << " from:" << request->sender_id();
196 return Dispatch(std::move(context), std::move(request));
197}

199// Dispatch the request if it is a heart beat message from other replica or a
200// cert notification from clients. Otherwise, forward to the worker.
201int ConsensusManager::Dispatch(std::unique_ptr<Context> context,
202 std::unique_ptr<Request> request) {
203 if (request->type() == Request::TYPE_HEART_BEAT) {
204 return ProcessHeartBeat(std::move(context), std::move(request));
205 }
206 return ConsensusCommit(std::move(context), std::move(request));
207}

209int ConsensusManager::ProcessHeartBeat(std::unique_ptr<Context> context,
210 std::unique_ptr<Request> request) {
211 std::vector<ReplicaInfo> replicas = GetReplicas();
212 HeartBeatInfo hb_info;
213 if (!hb_info.ParseFromString(request->data())) {
214 LOG(ERROR) << "parse replica info fail\n";
215 return -1;
216 }
217
218 LOG(INFO) << "receive public size:" << hb_info.public_keys().size()
219 << " primary:" << hb_info.primary()
220 << " version:" << hb_info.version()
221 << " from region:" << request->region_info().region_id();
222
223 if (request->region_info().region_id() ==
224 config_.GetConfigData().self_region_id()) {
225 if (config_.GetPublicKeyCertificateInfo()
226 .public_key()
227 .public_key_info()
228 .type() == CertificateKeyInfo::CLIENT) {
229 // TODO count 2f+1 before setting a new primary
230 SetPrimary(hb_info.primary(), hb_info.version());
231 }
232 }
233
234 int replica_num = 0;
235 // Update the public keys received from others.
236 for (const auto& public_key : hb_info.public_keys()) {
237 if (verifier_ && !verifier_->AddPublicKey(public_key)) {
238 LOG(ERROR) << "set public key fail from:"
239 << public_key.public_key_info().node_id();
240 continue;
241 }
242 if (request->region_info().region_id() !=
243 config_.GetConfigData().self_region_id()) {
244 // LOG(ERROR) << "key from other region:"
245 // << request->region_info().region_id();
246 continue;
247 }
248
249 ReplicaInfo info;
250 info.set_ip(public_key.public_key_info().ip());
251 info.set_port(public_key.public_key_info().port());
252 info.set_id(public_key.public_key_info().node_id());
253 if (info.ip().empty()) {
254 LOG(ERROR) << "public doesn't have ip, skip";
255 continue;
256 }
257 // Check whether there is a new replica joining.
258 // TODO notify new replica
259 if (public_key.public_key_info().type() == CertificateKeyInfo::REPLICA) {
260 replica_num++;
261 if (!ReplicaExisted(info, replicas)) {
262 // AddNewReplica(info);
263 }
264 } else {
265 if (!ReplicaExisted(info, clients_)) {
266 AddNewClient(info);
267 }
268 }
269 }
270 if (!is_ready_ && replica_num >= config_.GetMinDataReceiveNum()) {
271 LOG(ERROR) << "============ Server " << config_.GetSelfInfo().id()
272 << " is ready "
273 "=====================";
274 is_ready_ = true;
275 }
276 return 0;
277}

306void ConsensusManager::SendMessage(const google::protobuf::Message& message,
307 int64_t node_id) {
308 std::vector<ReplicaInfo> replicas = GetReplicas();
309 ReplicaInfo target_replica;
310 for (const auto& replica : replicas) {
311 if (replica.id() == node_id) {
312 target_replica = replica;
313 break;
314 }
315 }
316
317 if (target_replica.ip().empty()) {
318 return;
319 }
320
321 int ret = bc_client_->SendMessage(message, target_replica);
322 if (ret < 0) {
323 LOG(ERROR) << "broadcast request fail:";
324 }
325}
