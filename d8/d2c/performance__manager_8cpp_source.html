129// =================== response ========================
130// handle the response message. If receive f+1 commit messages, send back to the
131// user.
132int PerformanceManager::ProcessResponseMsg(std::unique_ptr<Context> context,
133 std::unique_ptr<Request> request) {
134 std::unique_ptr<Request> response;
135 std::string hash = request->hash();
136 int32_t primary_id = request->primary_id();
137 uint64_t seq = request->seq();
138 // Add the response message, and use the call back to collect the received
139 // messages.
140 // The callback will be triggered if it received f+1 messages.
141 if (request->ret() == -2) {
142 // LOG(INFO) << "get response fail:" << request->ret();
143 // send_num_--;
144 RemoveWaitingResponseRequest(hash);
145 return 0;
146 }
147 CollectorResultCode ret =
148 AddResponseMsg(context->signature, std::move(request),
149 [&](const Request& request,
150 const TransactionCollector::CollectorDataType*) {
151 response = std::make_unique<Request>(request);
152 return;
153 });
154
155 if (ret == CollectorResultCode::STATE_CHANGED) {
156 BatchUserResponse batch_response;
157 if (batch_response.ParseFromString(response->data())) {
158 if (seq > highest_seq_) {
159 highest_seq_ = seq;
160 if (highest_seq_primary_id_ != primary_id) {
161 system_info_->SetPrimary(primary_id);
162 highest_seq_primary_id_ = primary_id;
163 }
164 }
165 SendResponseToClient(batch_response);
166 RemoveWaitingResponseRequest(hash);
167 } else {
168 LOG(ERROR) << "parse response fail:";
169 }
170 }
171 return ret == CollectorResultCode::INVALID ? -2 : 0;
172}

174bool PerformanceManager::MayConsensusChangeStatus(
175 int type, int received_count, std::atomic<TransactionStatue>* status) {
176 switch (type) {
177 case Request::TYPE_RESPONSE:
178 // if receive f+1 response results, ack to the caller.
179 if (*status == TransactionStatue::None &&
180 config_.GetMinClientReceiveNum() <= received_count) {
181 TransactionStatue old_status = TransactionStatue::None;
182 return status->compare_exchange_strong(
183 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
184 std::memory_order_acq_rel);
185 }
186 break;
187 }
188 return false;
189}

191CollectorResultCode PerformanceManager::AddResponseMsg(
192 const SignatureInfo& signature, std::unique_ptr<Request> request,
193 std::function<void(const Request&,
194 const TransactionCollector::CollectorDataType*)>
195 response_call_back) {
196 if (request == nullptr) {
197 return CollectorResultCode::INVALID;
198 }
199
200 int type = request->type();
201 uint64_t seq = request->seq();
202 int resp_received_count = 0;
203 int ret = collector_pool_->GetCollector(seq)->AddRequest(
204 std::move(request), signature, false,
205 [&](const Request& request, int received_count,
206 TransactionCollector::CollectorDataType* data,
207 std::atomic<TransactionStatue>* status, bool force) {
208 if (MayConsensusChangeStatus(type, received_count, status)) {
209 resp_received_count = 1;
210 response_call_back(request, data);
211 }
212 });
213 if (ret != 0) {
214 return CollectorResultCode::INVALID;
215 }
216 if (resp_received_count > 0) {
217 collector_pool_->Update(seq);
218 return CollectorResultCode::STATE_CHANGED;
219 }
220 return CollectorResultCode::OK;
221}

223void PerformanceManager::SendResponseToClient(
224 const BatchUserResponse& batch_response) {
225 uint64_t create_time = batch_response.createtime();
226 uint64_t local_id = batch_response.local_id();
227 if (create_time > 0) {
228 uint64_t run_time = GetCurrentTime() - create_time;
229 global_stats_->AddLatency(run_time);
230 } else {
231 LOG(ERROR) << "seq:" << local_id << " no resp";
232 }
233 {
234 // std::lock_guard<std::mutex> lk(mutex_);
235 if (send_num_[batch_response.primary_id()] > 0) {
236 send_num_[batch_response.primary_id()]--;
237 }
238 }
239
240 if (config_.IsPerformanceRunning()) {
241 return;
242 }
243}

246int PerformanceManager::BatchProposeMsg() {
247 LOG(WARNING) << "batch wait time:" << config_.ClientBatchWaitTimeMS()
248 << " batch num:" << config_.ClientBatchNum()
249 << " max txn:" << config_.GetMaxProcessTxn();
250 std::vector<std::unique_ptr<QueueItem>> batch_req;
251 eval_ready_future_.get();
252 while (!stop_) {
253 // std::lock_guard<std::mutex> lk(mutex_);
254 if (send_num_[GetPrimary()] >= config_.GetMaxProcessTxn()) {
255 usleep(100000);
256 continue;
257 }
258 if (batch_req.size() < config_.ClientBatchNum()) {
259 std::unique_ptr<QueueItem> item =
260 batch_queue_.Pop(config_.ClientBatchWaitTimeMS());
261 if (item == nullptr) {
262 continue;
263 }
264 batch_req.push_back(std::move(item));
265 if (batch_req.size() < config_.ClientBatchNum()) {
266 continue;
267 }
268 }
269 int ret = DoBatch(batch_req);
270 batch_req.clear();
271 if (ret != 0) {
272 Response response;
273 response.set_result(Response::ERROR);
274 for (size_t i = 0; i < batch_req.size(); ++i) {
275 if (batch_req[i]->context && batch_req[i]->context->client) {
276 int ret = batch_req[i]->context->client->SendRawMessage(response);
277 if (ret) {
278 LOG(ERROR) << "send resp" << response.DebugString()
279 << " fail ret:" << ret;
280 }
281 }
282 }
283 }
284 }
285 return 0;
286}

288int PerformanceManager::DoBatch(
289 const std::vector<std::unique_ptr<QueueItem>>& batch_req) {
290 auto new_request =
291 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
292 if (new_request == nullptr) {
293 return -2;
294 }
295 std::vector<std::unique_ptr<Context>> context_list;
296
297 BatchUserRequest batch_request;
298 for (size_t i = 0; i < batch_req.size(); ++i) {
299 BatchUserRequest::UserRequest* req = batch_request.add_user_requests();
300 *req->mutable_request() = *batch_req[i]->user_request.get();
301 if (batch_req[i]->context) {
302 *req->mutable_signature() = batch_req[i]->context->signature;
303 }
304 req->set_id(i);
305 }
306
307 batch_request.set_createtime(GetCurrentTime());
308 batch_request.SerializeToString(new_request->mutable_data());
309 if (verifier_) {
310 auto signature_or = verifier_->SignMessage(new_request->data());
311 if (!signature_or.ok()) {
312 LOG(ERROR) << "Sign message fail";
313 return -2;
314 }
315 *new_request->mutable_data_signature() = *signature_or;
316 }
317
318 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
319 new_request->set_proxy_id(config_.GetSelfInfo().id());
320
321 replica_communicator_->SendMessage(*new_request, GetPrimary());
322 global_stats_->BroadCastMsg();
323 send_num_[GetPrimary()]++;
324 if (total_num_++ == 1000000) {
325 stop_ = true;
326 LOG(WARNING) << "total num is done:" << total_num_;
327 }
328 if (total_num_ % 10000 == 0) {
329 LOG(WARNING) << "total num is :" << total_num_;
330 }
331 global_stats_->IncClientCall();
332 AddWaitingResponseRequest(std::move(new_request));
333 return 0;
334}

336void PerformanceManager::AddWaitingResponseRequest(
337 std::unique_ptr<Request> request) {
338 if (!config_.GetConfigData().enable_viewchange()) {
339 return;
340 }
341 pm_lock_.lock();
342 uint64_t time = GetCurrentTime() + this->timeout_length_;
343 client_timeout_min_heap_.push(
344 PerformanceClientTimeout(request->hash(), time));
345 waiting_response_batches_.insert(
346 make_pair(request->hash(), std::move(request)));
347 pm_lock_.unlock();
348 sem_post(&request_sent_signal_);
349}

351void PerformanceManager::RemoveWaitingResponseRequest(std::string hash) {
352 if (!config_.GetConfigData().enable_viewchange()) {
353 return;
354 }
355 pm_lock_.lock();
356 if (waiting_response_batches_.find(hash) != waiting_response_batches_.end()) {
357 waiting_response_batches_.erase(waiting_response_batches_.find(hash));
358 }
359 pm_lock_.unlock();
360}

378void PerformanceManager::MonitoringClientTimeOut() {
379 while (!stop_) {
380 sem_wait(&request_sent_signal_);
381 pm_lock_.lock();
382 if (client_timeout_min_heap_.empty()) {
383 pm_lock_.unlock();
384 continue;
385 }
386 auto client_timeout = client_timeout_min_heap_.top();
387 client_timeout_min_heap_.pop();
388 pm_lock_.unlock();
389
390 if (client_timeout.timeout_time > GetCurrentTime()) {
391 usleep(client_timeout.timeout_time - GetCurrentTime());
392 }
393
394 if (CheckTimeOut(client_timeout.hash)) {
395 auto request = GetTimeOutRequest(client_timeout.hash);
396 if (request) {
397 LOG(ERROR) << "Client Request Timeout " << client_timeout.hash;
398 replica_communicator_->BroadCast(*request);
399 }
400 }
401 }
