39Transaction::Transaction(const Config& config, Wallet* wallet)
40 : config_(config), wallet_(wallet) {
41 tx_mempool_ = std::make_unique<TxMempool>();
42 for (const UTXO& trans : config_.genesis_transactions().transactions()) {
43 int64_t transaction_id = tx_mempool_->AddUTXO(trans);
44
45 for (const UTXOOut& output : trans.out()) {
46 int ret = wallet_->AddCoin(output.address(), output.value());
47 if (ret) {
48 LOG(ERROR) << "add coin fail";
49 assert(ret == 0);
50 }
51 }
52
53 LOG(ERROR) << "get genesis utxo:" << trans.DebugString()
54 << " transaction id:" << transaction_id;
55 }
56}

60int64_t Transaction::AddTransaction(const UTXO& utxo) {
61 absl::StatusOr<std::vector<UTXOOut>> ins_or = GetInput(utxo);
62 if (!ins_or.ok()) {
63 LOG(ERROR) << "get input fail:" << ins_or.status().message();
64 return -1;
65 }
66
67 if (!VerifyUTXO(utxo, *ins_or)) {
68 LOG(ERROR) << "transaction is not valid";
69 return -1;
70 }
71
72 if (AddCoin(utxo)) {
73 LOG(ERROR) << "add coin fail";
74 return -1;
75 }
76 return tx_mempool_->AddUTXO(utxo);
77}

88absl::StatusOr<std::vector<UTXOOut>> Transaction::GetInput(const UTXO& utxo) {
89 std::vector<UTXOOut> utxos;
90 if (utxo.in_size() == 0) {
91 if (utxo.address() == "0000") {
92 return utxos;
93 }
94 }
95
96 int64_t verify_nonce = 0;
97 std::string public_key;
98 for (const UTXOIn& input : utxo.in()) {
99 absl::StatusOr<UTXOOut> utxo_or =
100 tx_mempool_->GetUTXO(input.prev_id(), input.out_idx(), utxo.address());
101 if (!utxo_or.ok()) {
102 LOG(ERROR) << "get pre utxo id fail:" << input.prev_id();
103 return utxo_or.status();
104 }
105 utxos.push_back(*utxo_or);
106 if (public_key.empty()) {
107 public_key = (*utxo_or).pub_key();
108 }
109 verify_nonce += input.prev_id();
110 }
111
112 if (utxos.empty()) {
113 LOG(ERROR) << "no input";
114 return absl::InvalidArgumentError("Input invalid.");
115 }
116
117 if (utxo.sig().empty()) {
118 LOG(ERROR) << "utxo no sig, invalid";
119 return absl::InvalidArgumentError("Input invalid.");
120 }
121
122 LOG(ERROR) << "get public key:" << public_key << " nonce:" << verify_nonce;
123
124 bool valid = utils::ECDSAVerifyString(
125 utxo.address() + std::to_string(verify_nonce), public_key, utxo.sig());
126 if (!valid) {
127 LOG(ERROR) << "key not valid";
128 return absl::InvalidArgumentError("Key invalid.");
129 }
130
131 for (const UTXOOut& utxo : utxos) {
132 if (utxo.pub_key() != public_key) {
133 LOG(ERROR) << "public key not match";
134 return absl::InvalidArgumentError("Key invalid.");
135 }
136 }
137
138 return utxos;
139}

141bool Transaction::VerifyUTXO(const UTXO& utxo,
142 const std::vector<UTXOOut>& inputs) {
143 int64_t total_input_v = 0;
144 for (const UTXOOut& input : inputs) {
145 int64_t value = input.value();
146 total_input_v += value;
147 }
148
149 for (const UTXOOut& output : utxo.out()) {
150 total_input_v -= output.value();
151 }
152
153 if (total_input_v <= 0) {
154 LOG(ERROR) << "input is not enough";
155 return false;
156 }
157
158 return true;
159}

166int Transaction::AddCoin(const UTXO& utxo) {
167 int64_t total_value = 0;
168 for (const UTXOIn& input : utxo.in()) {
169 int64_t value = tx_mempool_->MarkSpend(input.prev_id(), input.out_idx(),
170 utxo.address());
171 assert(value >= 0);
172 total_value += value;
173 }
174
175 int ret = wallet_->AddCoin(utxo.address(), -total_value);
176 if (ret) {
177 LOG(ERROR) << "add coin fail";
178 return -1;
179 }
180
181 for (const UTXOOut& output : utxo.out()) {
182 int ret = wallet_->AddCoin(output.address(), output.value());
183 if (ret) {
184 LOG(ERROR) << "add coin fail";
185 return -1;
186 }
187 }
188 return 0;
189}
