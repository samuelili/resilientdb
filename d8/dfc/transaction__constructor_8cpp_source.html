40absl::StatusOr<std::string> TransactionConstructor::GetResponseData(
41 const Response& response) {
42 std::string hash_;
43 std::set<int64_t> hash_counter;
44 std::string resp_str;
45 for (const auto& each_resp : response.resp()) {
46 // Check signature
47 std::string hash = SignatureVerifier::CalculateHash(each_resp.data());
48
49 if (!hash_.empty() && hash != hash_) {
50 LOG(ERROR) << "hash not the same";
51 return absl::InvalidArgumentError("hash not match");
52 }
53 if (hash_.empty()) {
54 hash_ = hash;
55 resp_str = each_resp.data();
56 }
57 hash_counter.insert(each_resp.signature().node_id());
58 }
59 // LOG(INFO) << "recv hash:" << hash_counter.size()
60 // << " need:" << config_.GetMinClientReceiveNum()
61 // << " data len:" << resp_str.size();
62 if (hash_counter.size() >=
63 static_cast<size_t>(config_.GetMinClientReceiveNum())) {
64 return resp_str;
65 }
66 return absl::InvalidArgumentError("data not enough");
67}

76int TransactionConstructor::SendRequest(
77 const google::protobuf::Message& message,
78 google::protobuf::Message* response, Request::Type type) {
79 NetChannel::SetDestReplicaInfo(config_.GetReplicaInfos()[0]);
80 int ret = NetChannel::SendRequest(message, type, true);
81 if (ret == 0) {
82 std::string resp_str;
83 int ret = NetChannel::RecvRawMessageData(&resp_str);
84 if (ret >= 0) {
85 if (!response->ParseFromString(resp_str)) {
86 LOG(ERROR) << "parse response fail:" << resp_str.size();
87 return -2;
88 }
89 return 0;
90 }
91 }
92 return -1;
93}
