65// Handle the user request and send a pre-prepare message to others.
66// TODO if not a primary, redicet to the primary replica.
67int Commitment::ProcessNewRequest(std::unique_ptr<Context> context,
68 std::unique_ptr<Request> user_request) {
69 if (context == nullptr || context->signature.signature().empty()) {
70 LOG(ERROR) << "user request doesn't contain signature, reject";
71 return -2;
72 }
73
74 if (uint64_t seq =
75 duplicate_manager_->CheckIfExecuted(user_request->hash())) {
76 LOG(ERROR) << "This request is already executed with seq: " << seq;
77 user_request->set_seq(seq);
78 message_manager_->SendResponse(std::move(user_request));
79 return -2;
80 }
81
82 if (config_.GetSelfInfo().id() != message_manager_->GetCurrentPrimary()) {
83 // LOG(ERROR) << "current node is not primary. primary:"
84 // << message_manager_->GetCurrentPrimary()
85 // << " seq:" << user_request->seq()
86 // << " hash:" << user_request->hash();
87 replica_communicator_->SendMessage(*user_request,
88 message_manager_->GetCurrentPrimary());
89 {
90 std::lock_guard<std::mutex> lk(rc_mutex_);
91 request_complained_.push(
92 std::make_pair(std::move(context), std::move(user_request)));
93 }
94
95 return -3;
96 }
97
98 /*
99 if(SignatureVerifier::CalculateHash(user_request->data()) !=
100 user_request->hash()){ LOG(ERROR) << "the hash and data of the user request
101 don't match, reject"; return -2;
102 }
103 */
104
105 // check signatures
106 bool valid = verifier_->VerifyMessage(user_request->data(),
107 user_request->data_signature());
108 if (!valid) {
109 LOG(ERROR) << "request is not valid:"
110 << user_request->data_signature().DebugString();
111 LOG(ERROR) << " msg:" << user_request->data().size();
112 return -2;
113 }
114
115 if (pre_verify_func_ && !pre_verify_func_(*user_request)) {
116 LOG(ERROR) << " check by the user func fail";
117 return -2;
118 }
119
120 global_stats_->IncClientRequest();
121 if (duplicate_manager_->CheckAndAddProposed(user_request->hash())) {
122 LOG(ERROR) << "duplicate check fail:";
123 return -2;
124 }
125 auto seq = message_manager_->AssignNextSeq();
126
127 // Artificially make the primary stop proposing new trasactions.
128
129 if (!seq.ok()) {
130 duplicate_manager_->EraseProposed(user_request->hash());
131 global_stats_->SeqFail();
132 Request request;
133 request.set_type(Request::TYPE_RESPONSE);
134 request.set_sender_id(config_.GetSelfInfo().id());
135 request.set_proxy_id(user_request->proxy_id());
136 request.set_ret(-2);
137 request.set_hash(user_request->hash());
138
139 replica_communicator_->SendMessage(request, request.proxy_id());
140 return -2;
141 }
142
143 user_request->set_type(Request::TYPE_PRE_PREPARE);
144 user_request->set_current_view(message_manager_->GetCurrentView());
145 user_request->set_seq(*seq);
146 user_request->set_sender_id(config_.GetSelfInfo().id());
147 user_request->set_primary_id(config_.GetSelfInfo().id());
148
149 replica_communicator_->BroadCast(*user_request);
150
151 return 0;
152}

154// Receive the pre-prepare message from the primary.
155// TODO check whether the sender is the primary.
156int Commitment::ProcessProposeMsg(std::unique_ptr<Context> context,
157 std::unique_ptr<Request> request) {
158 if (context == nullptr || context->signature.signature().empty()) {
159 LOG(ERROR) << "user request doesn't contain signature, reject";
160 return -2;
161 }
162 if (request->is_recovery()) {
163 if (request->seq() >= message_manager_->GetNextSeq()) {
164 message_manager_->SetNextSeq(request->seq() + 1);
165 }
166 return message_manager_->AddConsensusMsg(context->signature,
167 std::move(request));
168 }
169
170 if (request->sender_id() != message_manager_->GetCurrentPrimary()) {
171 LOG(ERROR) << "the request is not from primary. sender:"
172 << request->sender_id() << " seq:" << request->seq();
173 return -2;
174 }
175
176 /*
177 if(request->hash() != "null" + std::to_string(request->seq())
178 && SignatureVerifier::CalculateHash(request->data()) != request->hash())
179 { LOG(ERROR) << "the hash and data of the request don't match, reject";
180 return -2;
181 }
182 */
183
184 if (request->sender_id() != config_.GetSelfInfo().id()) {
185 if (pre_verify_func_ && !pre_verify_func_(*request)) {
186 LOG(ERROR) << " check by the user func fail";
187 return -2;
188 }
189 BatchUserRequest batch_request;
190 batch_request.ParseFromString(request->data());
191 batch_request.clear_createtime();
192 std::string data;
193 batch_request.SerializeToString(&data);
194 // check signatures
195 bool valid =
196 verifier_->VerifyMessage(request->data(), request->data_signature());
197 if (!valid) {
198 LOG(ERROR) << "request is not valid:"
199 << request->data_signature().DebugString();
200 LOG(ERROR) << " msg:" << request->data().size();
201 return -2;
202 }
203 if (duplicate_manager_->CheckAndAddProposed(request->hash())) {
204 LOG(INFO) << "The request is already proposed, reject";
205 return -2;
206 }
207 }
208
209 global_stats_->IncPropose();
210 std::unique_ptr<Request> prepare_request = resdb::NewRequest(
211 Request::TYPE_PREPARE, *request, config_.GetSelfInfo().id());
212 prepare_request->clear_data();
213
214 // Add request to message_manager.
215 // If it has received enough same requests(2f+1), broadcast the prepare
216 // message.
217 CollectorResultCode ret =
218 message_manager_->AddConsensusMsg(context->signature, std::move(request));
219 if (ret == CollectorResultCode::STATE_CHANGED) {
220 replica_communicator_->BroadCast(*prepare_request);
221 }
222 return ret == CollectorResultCode::INVALID ? -2 : 0;
223}

225// If receive 2f+1 prepare message, broadcast a commit message.
226int Commitment::ProcessPrepareMsg(std::unique_ptr<Context> context,
227 std::unique_ptr<Request> request) {
228 if (context == nullptr || context->signature.signature().empty()) {
229 LOG(ERROR) << "user request doesn't contain signature, reject";
230 return -2;
231 }
232 if (request->is_recovery()) {
233 return message_manager_->AddConsensusMsg(context->signature,
234 std::move(request));
235 }
236 global_stats_->IncPrepare();
237 std::unique_ptr<Request> commit_request = resdb::NewRequest(
238 Request::TYPE_COMMIT, *request, config_.GetSelfInfo().id());
239 commit_request->mutable_data_signature()->Clear();
240 // Add request to message_manager.
241 // If it has received enough same requests(2f+1), broadcast the commit
242 // message.
243 uint64_t seq_ = request->seq();
244 CollectorResultCode ret =
245 message_manager_->AddConsensusMsg(context->signature, std::move(request));
246 if (ret == CollectorResultCode::STATE_CHANGED) {
247 if (message_manager_->GetHighestPreparedSeq() < seq_) {
248 message_manager_->SetHighestPreparedSeq(seq_);
249 }
250 // If need qc, sign the data
251 if (need_qc_ && verifier_) {
252 auto signature_or = verifier_->SignMessage(commit_request->hash());
253 if (!signature_or.ok()) {
254 LOG(ERROR) << "Sign message fail";
255 return -2;
256 }
257 *commit_request->mutable_data_signature() = *signature_or;
258 // LOG(ERROR) << "sign hash"
259 // << commit_request->data_signature().DebugString();
260 }
261 replica_communicator_->BroadCast(*commit_request);
262 }
263 return ret == CollectorResultCode::INVALID ? -2 : 0;
264}

266// If receive 2f+1 commit message, commit the request.
267int Commitment::ProcessCommitMsg(std::unique_ptr<Context> context,
268 std::unique_ptr<Request> request) {
269 if (context == nullptr || context->signature.signature().empty()) {
270 LOG(ERROR) << "user request doesn't contain signature, reject"
271 << " context:" << (context == nullptr);
272 return -2;
273 }
274 if (request->is_recovery()) {
275 return message_manager_->AddConsensusMsg(context->signature,
276 std::move(request));
277 }
278 global_stats_->IncCommit();
279 // Add request to message_manager.
280 // If it has received enough same requests(2f+1), message manager will
281 // commit the request.
282 CollectorResultCode ret =
283 message_manager_->AddConsensusMsg(context->signature, std::move(request));
284 return ret == CollectorResultCode::INVALID ? -2 : 0;
285}

288// If the transaction is executed, send back to the proxy.
289int Commitment::PostProcessExecutedMsg() {
290 while (!stop_) {
291 auto batch_resp = message_manager_->GetResponseMsg();
292 if (batch_resp == nullptr) {
293 continue;
294 }
295 Request request;
296 request.set_hash(batch_resp->hash());
297 request.set_seq(batch_resp->seq());
298 request.set_type(Request::TYPE_RESPONSE);
299 request.set_sender_id(config_.GetSelfInfo().id());
300 request.set_current_view(batch_resp->current_view());
301 request.set_proxy_id(batch_resp->proxy_id());
302 request.set_primary_id(batch_resp->primary_id());
303 // LOG(ERROR)<<"send back to proxy:"<<batch_resp->proxy_id();
304 batch_resp->SerializeToString(request.mutable_data());
305 replica_communicator_->SendMessage(request, request.proxy_id());
306 }
307 return 0;
308}
