111void DuplicateManager::UpdateRecentHash() {
112 uint64_t time = GetCurrentTime();
113 while (!IsStop()) {
114 time = time + frequency_useconds_;
115 auto sleep_time = time - GetCurrentTime();
116 usleep(sleep_time);
117 while (true) {
118 std::lock_guard<std::mutex> lk(prop_mutex_);
119 if (!proposed_hash_time_queue_.empty()) {
120 auto it = proposed_hash_time_queue_.front();
121 if (it.second + window_useconds_ < time) {
122 proposed_hash_time_queue_.pop();
123 proposed_hash_set_.erase(it.first);
124 } else {
125 break;
126 }
127 } else {
128 break;
129 }
130 }
131
132 while (true) {
133 std::lock_guard<std::mutex> lk(exec_mutex_);
134 if (!executed_hash_time_queue_.empty()) {
135 auto it = executed_hash_time_queue_.front();
136 if (it.second + window_useconds_ < time) {
137 executed_hash_time_queue_.pop();
138 executed_hash_set_.erase(it.first);
139 executed_hash_seq_.erase(it.first);
140 } else {
141 break;
142 }
143 } else {
144 break;
145 }
146 }
147 }
