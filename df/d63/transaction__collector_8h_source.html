51template <typename T>
52class AtomicUniquePtr {
53 public:
54 AtomicUniquePtr() : v_(0) {}
55 bool Set(std::unique_ptr<T>& new_ptr) {
56 int old_v = 0;
57 bool res = v_.compare_exchange_strong(old_v, 1, std::memory_order_acq_rel,
58 std::memory_order_acq_rel);
59 if (!res) {
60 return false;
61 }
62 ptr_ = std::move(new_ptr);
63 v_ = 2;
64 return true;
65 }
66
67 T* Reference() {
68 int v = v_.load(std::memory_order_acq_rel);
69 if (v <= 1) {
70 return nullptr;
71 }
72 return ptr_.get();
73 }
74
75 void Clear() {
76 v_ = 0;
77 ptr_ = nullptr;
78 }
79
80 private:
81 std::unique_ptr<T> ptr_;
82 std::atomic<int> v_;
83};
