84// Connect to the server if not connected and send data.
85int NetChannel::SendFromKeepAlive(const std::string& data) {
86 for (int i = 0; i < max_retry_time_; ++i) {
87 if (!long_connecting_) {
88 if (Connect() == 0) {
89 long_connecting_ = true;
90 }
91 if (!long_connecting_) {
92 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
93 continue;
94 }
95 }
96 int ret = SendDataInternal(data);
97 if (ret >= 0) {
98 return ret;
99 }
100 long_connecting_ = false;
101 }
102 return -1;
103}

105int NetChannel::Send(const std::string& data) {
106 if (long_connect_tion_) {
107 return SendFromKeepAlive(data);
108 }
109 for (int i = 0; i < max_retry_time_; ++i) {
110 if (!connected_) {
111 if (Connect()) {
112 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
113 continue;
114 }
115 }
116 int ret = SendDataInternal(data);
117 if (ret >= 0) {
118 return ret;
119 } else if (ret == -1) {
120 LOG(ERROR) << "send data fail:" << ip_ << " port:" << port_;
121 }
122 }
123 return -1;
124}

136// Sign the message if verifier has been provied and send the message to the
137// server.
138std::string NetChannel::GetRawMessageString(
139 const google::protobuf::Message& message, SignatureVerifier* verifier) {
140 ResDBMessage sig_message;
141 if (!message.SerializeToString(sig_message.mutable_data())) {
142 return "";
143 }
144
145 if (verifier != nullptr) {
146 auto signature_or = verifier->SignMessage(sig_message.data());
147 if (!signature_or.ok()) {
148 LOG(ERROR) << "Sign message fail";
149 return "";
150 }
151 sig_message.mutable_signature()->Swap(&(*signature_or));
152 } else {
153 // LOG(ERROR) << " no verifier";
154 }
155
156 std::string message_str;
157 if (!sig_message.SerializeToString(&message_str)) {
158 return "";
159 }
160 return message_str;
161}

171int NetChannel::SendRawMessage(const google::protobuf::Message& message) {
172 ResDBMessage sig_message;
173 if (!message.SerializeToString(sig_message.mutable_data())) {
174 return -1;
175 }
176
177 if (verifier_ != nullptr) {
178 auto signature_or = verifier_->SignMessage(sig_message.data());
179 if (!signature_or.ok()) {
180 LOG(ERROR) << "Sign message fail";
181 return -1;
182 }
183 sig_message.mutable_signature()->Swap(&(*signature_or));
184 }
185 std::string message_str;
186 if (!sig_message.SerializeToString(&message_str)) {
187 return -1;
188 }
189 return Send(message_str);
190}

192int NetChannel::RecvRawMessageStr(std::string* message) {

197 }

201 return -1;
202 }
203 *message = sig_message.data();

207int NetChannel::RecvRawMessage(google::protobuf::Message* message) {
208 std::string resp_data;
209 int ret = Recv(&resp_data);
210 if (ret < 0) {
211 LOG(ERROR) << "recv fail:" << ret;
212 return -1;
213 }
214 ResDBMessage sig_message;
215 if (!sig_message.ParseFromString(resp_data)) {
216 LOG(ERROR) << "parse sig msg fail";
217 return -1;
218 }
219 if (!message->ParseFromString(sig_message.data())) {
220 LOG(ERROR) << "parse response msg fail";
221 return -1;
222 }
223 return 0;
224}
