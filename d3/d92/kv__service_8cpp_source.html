46std::unique_ptr<ChainState> NewState(const std::string& cert_file,
47 const ResConfigData& config_data) {
48 std::unique_ptr<Storage> storage = nullptr;
49
50#ifdef ENABLE_ROCKSDB
51 storage = NewResRocksDB(cert_file.c_str(), config_data);
52 LOG(INFO) << "use rocksdb storage.";
53#endif
54
55#ifdef ENABLE_LEVELDB
56 storage = NewResLevelDB(cert_file.c_str(), config_data);
57 LOG(INFO) << "use leveldb storage.";
58#endif
59 std::unique_ptr<ChainState> state =
60 std::make_unique<ChainState>(std::move(storage));
61 return state;
62}
