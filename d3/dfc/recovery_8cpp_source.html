111void Recovery::UpdateStableCheckPoint() {
112 if (checkpoint_ == nullptr) {
113 return;
114 }
115 while (!stop_) {
116 int64_t latest_ckpt = checkpoint_->GetStableCheckpoint();
117 LOG(ERROR) << "get stable ckpt:" << latest_ckpt;
118 if (last_ckpt_ == latest_ckpt) {
119 sleep(recovery_ckpt_time_s_);
120 continue;
121 }
122 last_ckpt_ = latest_ckpt;
123 FinishFile(latest_ckpt);
124 }

127void Recovery::GetLastFile() {
128 std::string dir = std::filesystem::path(file_path_).parent_path();
129 last_ckpt_ = -1;
130 int m_time_s = 0;
131 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
132 std::string dir = std::filesystem::path(entry.path()).parent_path();
133 std::string file_name = std::filesystem::path(entry.path()).stem();
134 std::string ext = std::filesystem::path(entry.path()).extension();
135 if (ext != ".log") continue;
136 int pos = file_name.rfind("_");
137 int64_t ckpt = std::stoll(file_name.substr(pos + 1));
138 int max_seq_pos = file_name.rfind("_", pos - 1);
139 int min_seq_pos = file_name.rfind("_", max_seq_pos - 1);
140 int time_pos = file_name.rfind("_", min_seq_pos - 1);
141
142 int64_t min_seq = std::stoll(
143 file_name.substr(min_seq_pos + 1, max_seq_pos - min_seq_pos - 1));
144
145 int64_t time_s =
146 std::stoll(file_name.substr(time_pos + 1, min_seq_pos - time_pos - 1));
147 if (min_seq == -1) {
148 if (last_ckpt_ == -1 || m_time_s < time_s) {
149 file_path_ = entry.path();
150 last_ckpt_ = ckpt;
151 m_time_s = time_s;
152 LOG(ERROR) << "get last path:" << file_name << " min:" << min_seq;
153 }
154 }
155 }
156 if (last_ckpt_ == -1) {
157 last_ckpt_ = 0;
158 file_path_ = GenerateFile(last_ckpt_, -1, -1);
159 }
160}

175void Recovery::FinishFile(int64_t seq) {
176 std::unique_lock<std::mutex> lk(mutex_);
177 Flush();
178 if (storage_) {
179 if (!storage_->Flush()) {
180 return;
181 }
182 }
183 std::string new_file_path = GenerateFile(seq, min_seq_, max_seq_);
184 close(fd_);
185
186 min_seq_ = -1;
187 max_seq_ = -1;
188
189 std::rename(file_path_.c_str(), new_file_path.c_str());
190
191 LOG(INFO) << "rename:" << file_path_ << " to:" << new_file_path;
192 std::string next_file_path = GenerateFile(seq, -1, -1);
193 file_path_ = next_file_path;
194
195 OpenFile(file_path_);
196}

198void Recovery::SwitchFile(const std::string& file_path) {
199 std::unique_lock<std::mutex> lk(mutex_);
200
201 min_seq_ = -1;
202 max_seq_ = -1;
203
204 ReadLogsFromFiles(
205 file_path, 0, 0, [&](const SystemInfoData& data) {},
206 [&](std::unique_ptr<Context> context, std::unique_ptr<Request> request) {
207 min_seq_ == -1
208 ? min_seq_ = request->seq()
209 : std::min(min_seq_, static_cast<int64_t>(request->seq()));
210 max_seq_ = std::max(max_seq_, static_cast<int64_t>(request->seq()));
211 });
212
213 OpenFile(file_path);
214 LOG(INFO) << "switch to file:" << file_path << " seq:"
215 << "[" << min_seq_ << "," << max_seq_ << "]";

219 if (fd_ >= 0) {
220 close(fd_);
221 }
222 fd_ = open(path.c_str(), O_CREAT | O_WRONLY, 0666);
223 if (fd_ < 0) {

298std::vector<std::unique_ptr<Recovery::RecoveryData>> Recovery::ParseData(
299 const std::string& data) {
300 std::vector<std::unique_ptr<RecoveryData>> request_list;
301
302 std::vector<std::string> data_list;
303 int pos = 0;
304 while (pos < data.size()) {
305 size_t len;
306 memcpy(&len, data.c_str() + pos, sizeof(len));
307 pos += sizeof(len);
308
309 std::string item = data.substr(pos, len);
310 pos += len;
311 data_list.push_back(item);
312 }
313
314 for (size_t i = 0; i < data_list.size(); i += 2) {
315 std::unique_ptr<RecoveryData> recovery_data =
316 std::make_unique<RecoveryData>();
317 recovery_data->request = std::make_unique<Request>();
318 recovery_data->context = std::make_unique<Context>();
319
320 if (!recovery_data->request->ParseFromString(data_list[i])) {
321 LOG(ERROR) << "Parse from data fail";
322 break;
323 }
324
325 if (!recovery_data->context->signature.ParseFromString(data_list[i + 1])) {
326 LOG(ERROR) << "Parse from data fail";
327 break;
328 }
329
330 request_list.push_back(std::move(recovery_data));
331 }
332 return request_list;
333}

391Recovery::GetRecoveryFiles() {
392 std::string dir = std::filesystem::path(file_path_).parent_path();
393 int64_t last_ckpt = 0;
394 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
395 std::string dir = std::filesystem::path(entry.path()).parent_path();
396 std::string file_name = std::filesystem::path(entry.path()).stem();
397 std::string ext = std::filesystem::path(entry.path()).extension();
398 if (ext != ".log") continue;
399 int pos = file_name.rfind("_");
400 int64_t ckpt = std::stoll(file_name.substr(pos + 1));
401 if (ckpt > last_ckpt) {
402 last_ckpt = ckpt;
403 }
404 }
405 std::vector<std::pair<int64_t, std::string>> list;
406
407 std::vector<std::pair<int64_t, std::string>> e_list;
408
409 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
410 std::string dir = std::filesystem::path(entry.path()).parent_path();
411 std::string file_name = std::filesystem::path(entry.path()).stem();
412 std::string ext = std::filesystem::path(entry.path()).extension();
413 if (ext != ".log") continue;
414 int pos = file_name.rfind("_");
415
416 int max_seq_pos = file_name.rfind("_", pos - 1);
417 int64_t max_seq =
418 std::stoll(file_name.substr(max_seq_pos + 1, pos - max_seq_pos - 1));
419
420 int min_seq_pos = file_name.rfind("_", max_seq_pos - 1);
421 int64_t min_seq = std::stoll(
422 file_name.substr(min_seq_pos + 1, max_seq_pos - min_seq_pos - 1));
423
424 int time_pos = file_name.rfind("_", min_seq_pos - 1);
425 int64_t time =
426 std::stoll(file_name.substr(time_pos + 1, min_seq_pos - time_pos - 1));
427
428 if (min_seq == -1) {
429 e_list.push_back(std::make_pair(time, entry.path()));
430 } else if ((min_seq <= last_ckpt && max_seq >= last_ckpt)) {
431 list.push_back(std::make_pair(time, entry.path()));
432 }
433 }
434
435 sort(e_list.begin(), e_list.end());
436 list.push_back(e_list.back());
437
438 sort(list.begin(), list.end());
439 return std::make_pair(list, last_ckpt);
440}

459void Recovery::ReadLogsFromFiles(
460 const std::string& path, int64_t ckpt, int file_idx,
461 std::function<void(const SystemInfoData& data)> system_callback,
462 std::function<void(std::unique_ptr<Context> context,
463 std::unique_ptr<Request> request)>
464 call_back) {
465 int fd = open(path.c_str(), O_CREAT | O_RDONLY, 0666);
466 if (fd < 0) {
467 LOG(ERROR) << " open file fail:" << path;
468 }
469 LOG(INFO) << "read logs:" << path << " pos:" << lseek(fd, 0, SEEK_CUR);
470 assert(fd >= 0);
471
472 size_t data_len = 0;
473 Read(fd, sizeof(data_len), reinterpret_cast<char*>(&data_len));
474 {
475 std::string data;
476 char* buf = new char[data_len];
477 if (!Read(fd, data_len, buf)) {
478 LOG(ERROR) << "Read system info fail";
479 return;
480 }
481 data = std::string(buf, data_len);
482 delete buf;
483 std::vector<std::string> data_list = ParseRawData(data);
484
485 SystemInfoData info;
486 if (data_list.empty() || !info.ParseFromString(data_list[0])) {
487 LOG(ERROR) << "parse info fail:" << data.size();
488 return;
489 }
490 LOG(INFO) << "read system info:" << info.DebugString();
491 if (file_idx == 0) {
492 system_callback(info);
493 }
494 }
495
496 std::vector<std::unique_ptr<RecoveryData>> request_list;
497
498 while (Read(fd, sizeof(data_len), reinterpret_cast<char*>(&data_len))) {
499 std::string data;
500 char* buf = new char[data_len];
501 if (!Read(fd, data_len, buf)) {
502 LOG(ERROR) << "Read data log fail";
503 break;
504 }
505 data = std::string(buf, data_len);
506 delete buf;
507
508 std::vector<std::unique_ptr<RecoveryData>> list = ParseData(data);
509 if (list.size() == 0) {
510 request_list.clear();
511 break;
512 }
513 for (auto& l : list) {
514 request_list.push_back(std::move(l));
515 }
516 }
517 if (request_list.size() == 0) {
518 ftruncate(fd, 0);
519 }
520 for (std::unique_ptr<RecoveryData>& recovery_data : request_list) {
521 if (ckpt < recovery_data->request->seq()) {
522 recovery_data->request->set_is_recovery(true);
523 call_back(std::move(recovery_data->context),
524 std::move(recovery_data->request));
525 }
526 }
527
528 LOG(INFO) << "read log from files:" << path << " done";
529
530 close(fd);
531}
