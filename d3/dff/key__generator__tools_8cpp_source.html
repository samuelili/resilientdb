35void WriteKey(const KeyInfo& key, const std::string& file_name) {
36 std::string str;
37 assert(key.SerializeToString(&str));
38
39 printf("save key to path %s\n", file_name.c_str());
40 int fd = open(file_name.c_str(), O_WRONLY | O_CREAT, 0666);
41 if (fd < 0) {
42 printf("open file fail %s\n", strerror(errno));
43 }
44 assert(fd >= 0);
45 write(fd, str.c_str(), str.size());
46 close(fd);
47}
