34 int i = 0;

41LockFreeCollectorPool::LockFreeCollectorPool(const std::string& name,
42 uint32_t size,
43 TransactionExecutor* executor,
44 bool enable_viewchange)
45 : name_(name),
46 capacity_(GetCapacity(size * 2)),
47 mask_((capacity_ << 1) - 1),
48 executor_(executor),
49 enable_viewchange_(enable_viewchange) {
50 collector_.resize(capacity_ << 1);
51 for (size_t i = 0; i < (capacity_ << 1); ++i) {
52 collector_[i] = std::make_unique<TransactionCollector>(i, executor_,
53 enable_viewchange_);
54 }
55 LOG(ERROR) << "name:" << name_ << " create pool done. capacity:" << capacity_
56 << " enable viewchange:" << enable_viewchange_ << " done";
57}

59void LockFreeCollectorPool::Update(uint64_t seq) {
60 uint32_t idx = seq & mask_;
61 if (collector_[idx]->Seq() != seq) {
62 LOG(ERROR) << "seq not match, skip update:" << seq;
63 return;
64 }
65 // LOG(ERROR)<<" update:"<<(idx^capacity_)<<" seq:"<<seq+capacity_<<
66 // " cap:"<<capacity_<<" update seq:"<<seq;
67 collector_[idx ^ capacity_] = std::make_unique<TransactionCollector>(
68 seq + capacity_, executor_, enable_viewchange_);
69}

71TransactionCollector* LockFreeCollectorPool::GetCollector(uint64_t seq) {
72 uint32_t idx = seq & mask_;
73 return collector_[idx].get();
