54void Acceptor::Run() {
55 LOG(ERROR) << "server:" << config_.GetSelfInfo().id() << " start running";
56
57 LockFreeQueue<Socket> socket_queue("server");
58 std::vector<std::thread> threads;
59 int woker_num = config_.GetInputWorkerNum();
60 for (int i = 0; i < woker_num; ++i) {
61 threads.push_back(std::thread([&]() {
62 while (IsRunning()) {
63 auto client_socket = socket_queue.Pop();
64 if (client_socket == nullptr) {
65 continue;
66 }
67 std::unique_ptr<DataInfo> request_info = std::make_unique<DataInfo>();
68 int ret =
69 client_socket->Recv(&request_info->buff, &request_info->data_len);
70 if (ret <= 0) {
71 continue;
72 }
73 std::unique_ptr<QueueItem> item = std::make_unique<QueueItem>();
74 item->socket = std::move(client_socket);
75 item->data = std::move(request_info);
76 global_stats_->ServerCall();
77 input_queue_->Push(std::move(item));
78 }
79 }));
80 }
81
82 while (IsRunning()) {
83 auto client_socket = socket_->Accept();
84 if (client_socket == nullptr) {
85 continue;
86 }
87 socket_queue.Push(std::move(client_socket));
88 }
89
90 for (auto& th : threads) {
91 if (th.joinable()) {
92 th.join();
93 }
94 }
95}
