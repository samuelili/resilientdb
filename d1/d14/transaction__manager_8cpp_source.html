44std::unique_ptr<BatchUserResponse> TransactionManager::ExecuteBatch(
45 const BatchUserRequest& request) {
46 std::unique_ptr<BatchUserResponse> batch_response =
47 std::make_unique<BatchUserResponse>();
48 for (auto& sub_request : request.user_requests()) {
49 std::unique_ptr<std::string> response =
50 ExecuteData(sub_request.request().data());
51 if (response == nullptr) {
52 response = std::make_unique<std::string>();
53 }
54 batch_response->add_response()->swap(*response);
55 }
56
57 return batch_response;
58}
