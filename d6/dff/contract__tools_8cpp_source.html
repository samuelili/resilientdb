80int main(int argc, char** argv) {
81 if (argc < 3) {
82 printf("<cmd> -c [config]\n");
83 return 0;
84 }
85
86 std::string cmd = argv[1];
87 std::string caller_address, contract_name, contract_path, params,
88 contract_address, func_name;
89 int c;
90 std::string client_config_file;
91 while ((c = getopt(argc, argv, "m:c:a:n:p:h:f:s:")) != -1) {
92 switch (c) {
93 case 'm':
94 caller_address = optarg;
95 break;
96 case 'c':
97 client_config_file = optarg;
98 break;
99 case 'n':
100 contract_name = optarg;
101 break;
102 case 'f':
103 func_name = optarg;
104 break;
105 case 'p':
106 contract_path = optarg;
107 break;
108 case 'a':
109 params = optarg;
110 break;
111 case 's':
112 contract_address = optarg;
113 break;
114 case 'h':
115 ShowUsage();
116 break;
117 }
118 }
119
120 printf("cmd = %s config path = %s\n", cmd.c_str(),
121 client_config_file.c_str());
122 ResDBConfig config = GenerateResDBConfig(client_config_file);
123 config.SetClientTimeoutMs(100000);
124
125 ContractClient client(config);
126
127 if (cmd == "create") {
128 CreateAccount(&client);
129 } else if (cmd == "deploy") {
130 std::vector<std::string> init_params;
131 boost::split(init_params, params, boost::is_any_of(","));
132
133 DeployContract(&client, caller_address, contract_name, contract_path,
134 init_params);
135 } else if (cmd == "execute") {
136 printf(
137 "execute\n caller address:%s\n contract address: %s\n func: %s\n "
138 "params:%s\n",
139 caller_address.c_str(), contract_address.c_str(), func_name.c_str(),
140 params.c_str());
141 std::vector<std::string> func_params;
142 boost::split(func_params, params, boost::is_any_of(","));
143
144 ExecuteContract(&client, caller_address, contract_address, func_name,
145 func_params);
146 }
147}
