81class ResDBTest : public Test {
82 public:
83 void StartServer() {}

85 void StartOneServer(int server_id) {
86 ResDBConfig config = GetConfig(server_id);
87 config.SetTestMode(true);
88 auto executor = std::make_unique<ResDBTestManager>(config);
89 executors_.push_back(executor.get());
90 server_.push_back(std::make_unique<ServiceNetwork>(
91 config,
92 std::make_unique<ConsensusManagerPBFT>(config, std::move(executor))));
93 server_thread_.push_back(std::thread(
94 [&](ServiceNetwork* server) { server->Run(); }, server_.back().get()));
95 }
96
97 void WaitAllServerStarted() {
98 for (auto& server : server_) {
99 while (true) {
100 std::unique_lock<std::mutex> lk(mutex_);
101 cv_.wait_for(lk, std::chrono::microseconds(1000),
102 [&] { return server->ServiceIsReady(); });
103 if (server->ServiceIsReady()) {
104 break;
105 }
106 }
107 }
108 }
109
110 void WaitExecutorDone(int received_num) {
111 for (auto executor : executors_) {
112 while (executor->GetSeqs().size() < received_num) {
113 usleep(10000);
114 }
115 }
116 }
117
118 void Stop() {
119 for (auto& server : server_) {
120 server->Stop();
121 }
122 for (auto& th : server_thread_) {
123 th.join();
124 }
125 server_.clear();
126 }
