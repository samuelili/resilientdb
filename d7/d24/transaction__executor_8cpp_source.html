93int TransactionExecutor::Commit(std::unique_ptr<Request> message) {
94 global_stats_->IncPendingExecute();
95 if (transaction_manager_ && transaction_manager_->IsOutOfOrder()) {
96 // LOG(ERROR)<<"add out of order exe:"<<message->seq()<<" from
97 // proxy:"<<message->proxy_id();
98 std::unique_ptr<Request> msg = std::make_unique<Request>(*message);
99 execute_OOO_queue_.Push(std::move(message));
100 commit_queue_.Push(std::move(msg));
101 } else {
102 commit_queue_.Push(std::move(message));
103 }

111std::unique_ptr<Request> TransactionExecutor::GetNextData() {
112 if (candidates_.empty() || candidates_.begin()->first != next_execute_seq_) {
113 return nullptr;
114 }
115 auto res = std::move(candidates_.begin()->second);
116 if (pre_exec_func_) {
117 pre_exec_func_(res.get());
118 }
119 candidates_.erase(candidates_.begin());
120 return res;
121}

123void TransactionExecutor::OrderMessage() {
124 while (!IsStop()) {
125 auto message = commit_queue_.Pop();
126 if (message != nullptr) {
127 global_stats_->IncExecute();
128 uint64_t seq = message->seq();
129 if (next_execute_seq_ > seq) {
130 // LOG(INFO) << "request seq:" << seq << " has been executed"
131 // << " next seq:" << next_execute_seq_;
132 continue;
133 }
134
135 AddNewData(std::move(message));
136 }
137
138 while (!IsStop()) {
139 std::unique_ptr<Request> message = GetNextData();
140 if (message == nullptr) {
141 break;
142 }
143 execute_queue_.Push(std::move(message));
144 next_execute_seq_++;
145 if (seq_update_notify_func_) {
146 seq_update_notify_func_(next_execute_seq_);
147 }
148 }
149 }
150 return;
151}

153void TransactionExecutor::ExecuteMessage() {
154 while (!IsStop()) {
155 auto message = execute_queue_.Pop();
156 if (message == nullptr) {
157 continue;
158 }
159 bool need_execute = true;
160 if (transaction_manager_ && transaction_manager_->IsOutOfOrder()) {
161 need_execute = false;
162 }
163 Execute(std::move(message), need_execute);
164 }
165}

167void TransactionExecutor::ExecuteMessageOutOfOrder() {

170 if (message == nullptr) {

205void TransactionExecutor::Execute(std::unique_ptr<Request> request,
206 bool need_execute) {
207 // Execute the request, then send the response back to the user.
208 BatchUserRequest batch_request;
209 if (!batch_request.ParseFromString(request->data())) {
210 LOG(ERROR) << "parse data fail";
211 }
212 batch_request.set_seq(request->seq());
213 batch_request.set_hash(request->hash());
214 batch_request.set_proxy_id(request->proxy_id());
215 if (request->has_committed_certs()) {
216 *batch_request.mutable_committed_certs() = request->committed_certs();
217 }
218
219 // LOG(INFO) << " get request batch size:"
220 // << batch_request.user_requests_size()<<" proxy
221 // id:"<<request->proxy_id()<<" need execute:"<<need_execute;
222 // std::unique_ptr<BatchUserResponse> batch_response =
223 // std::make_unique<BatchUserResponse>();
224
225 std::unique_ptr<BatchUserResponse> response;
226 if (transaction_manager_ && need_execute) {
227 response = transaction_manager_->ExecuteBatch(batch_request);
228 }
229
230 if (duplicate_manager_) {
231 duplicate_manager_->AddExecuted(batch_request.hash(), batch_request.seq());
232 }
233
234 global_stats_->IncTotalRequest(batch_request.user_requests_size());
235 if (response == nullptr) {
236 response = std::make_unique<BatchUserResponse>();
237 }
238
239 response->set_createtime(batch_request.createtime());
240 response->set_local_id(batch_request.local_id());
241 response->set_hash(batch_request.hash());
242
243 post_exec_func_(std::move(request), std::move(response));
244
245 global_stats_->IncExecuteDone();
246}
