43KeyInfo ReadKey(const std::string& file_name) {
44 int fd = open(file_name.c_str(), O_RDONLY, 0666);
45 if (fd < 0) {
46 LOG(ERROR) << "open file:" << file_name << " fail:" << strerror(errno);
47 }
48 assert(fd >= 0);
49
50 std::string res;
51 int read_len = 0;
52 char tmp[1024];
53 while (true) {
54 read_len = read(fd, tmp, sizeof(tmp));
55 if (read_len <= 0) {
56 break;
57 }
58 res.append(tmp, read_len);
59 }
60 close(fd);
61 KeyInfo key;
62 assert(key.ParseFromString(res));
63 return key;
64}

99ResConfigData ReadConfigFromFile(const std::string& file_name) {
100 std::stringstream json_data;
101 std::ifstream infile(file_name.c_str());
102 json_data << infile.rdbuf();
103
104 ResConfigData config_data;
105 JsonParseOptions options;
106 auto status = JsonStringToMessage(json_data.str(), &config_data, options);
107 if (!status.ok()) {
108 LOG(ERROR) << "parse json :" << file_name << " fail:" << status.message();
109 }
110 assert(status.ok());
111 return config_data;
112}

114std::vector<ReplicaInfo> ReadConfig(const std::string& file_name) {
115 std::vector<ReplicaInfo> replicas;
116 std::string line;
117 std::ifstream infile(file_name.c_str());
118 int id;
119 std::string ip;
120 int port;
121 while (infile >> id >> ip >> port) {
122 replicas.push_back(GenerateReplicaInfo(id, ip, port));
123 }
124 if (replicas.size() == 0) {
125 LOG(ERROR) << "read config:" << file_name << " fail.";
126 assert(replicas.size() > 0);
127 }
128 return replicas;
129}

131std::unique_ptr<ResDBConfig> GenerateResDBConfig(
132 const std::string& config_file, const std::string& private_key_file,
133 const std::string& cert_file, std::optional<ReplicaInfo> self_info,
134 std::optional<ConfigGenFunc> gen_func) {
135 ResConfigData config_data = ReadConfigFromFile(config_file);
136 KeyInfo private_key = ReadKey(private_key_file);
137 CertificateInfo cert_info = ReadCert(cert_file);
138
139 LOG(ERROR) << "private key:" << private_key.DebugString()
140 << " cert:" << cert_info.DebugString();
141 if (!self_info.has_value()) {
142 self_info = ReplicaInfo();
143 }
144
145 (*self_info).set_id(cert_info.public_key().public_key_info().node_id());
146 (*self_info).set_ip(cert_info.public_key().public_key_info().ip());
147 (*self_info).set_port(cert_info.public_key().public_key_info().port());
148
149 *(*self_info).mutable_certificate_info() = cert_info;
150
151 if (gen_func.has_value()) {
152 return (*gen_func)(config_data, self_info.value(), private_key, cert_info);
153 }
154 return std::make_unique<ResDBConfig>(config_data, self_info.value(),
155 private_key, cert_info);
156}
