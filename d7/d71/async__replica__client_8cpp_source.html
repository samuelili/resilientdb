44int AsyncReplicaClient::SendMessage(const std::string& data) {
45 queue_.Push(std::make_unique<std::string>(data));
46 if (!in_process_.load()) {
47 bool old_value = false;
48 if (in_process_.compare_exchange_strong(old_value, true,
49 std::memory_order_acq_rel,
50 std::memory_order_acq_rel)) {
51 OnSendNewMessage();
52 }
53 }
54 return 0;
55}

57void AsyncReplicaClient::OnSendNewMessage() {
58 std::unique_ptr<std::string> data = queue_.Pop(0);
59 if (data == nullptr || data->empty()) {
60 in_process_ = false;
61 return;
62 }
63 pending_data_ = std::move(data);
64 status_ = 0;
65 OnSendMessage();
66}

68void AsyncReplicaClient::OnSendMessage() {
69 if (status_ == 0) {
70 data_size_ = pending_data_->size();
71 sending_data_size_ = sizeof(data_size_);
72 sending_data_ptr_ = reinterpret_cast<char*>(&data_size_);
73 sending_data_idx_ = 0;
74 status_ = 1;
75 OnSend();
76 } else if (status_ == 1) {
77 sending_data_size_ = data_size_;
78 sending_data_ptr_ = pending_data_->c_str();
79 sending_data_idx_ = 0;
80 status_ = 2;
81 OnSend();
82 } else {
83 status_ = 0;
84 OnSendNewMessage();
85 }
86}

88void AsyncReplicaClient::OnSend() {
89 socket_.async_write_some(
90 boost::asio::buffer(sending_data_ptr_ + sending_data_idx_,
91 sending_data_size_ - sending_data_idx_),
92 [&](const boost::system::error_code& error, size_t send_size) {
93 if (error) {
94 ReConnect();
95 } else {
96 sending_data_idx_ += send_size;
97 if (sending_data_idx_ >= sending_data_size_) {
98 OnSendMessage();
99 } else {
100 OnSend();
101 }
102 }
103 });
104}

106void AsyncReplicaClient::ReConnect() {
107 socket_.async_connect(endpoint_, [&](const boost::system::error_code& error) {
108 if (!error) {
109 status_ = 0;
110 OnSendMessage();
111 } else {
112 usleep(10000);
113 ReConnect();
114 }
115 });
116}
