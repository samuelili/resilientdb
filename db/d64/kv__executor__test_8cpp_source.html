51 int Set(const std::string& key, const std::string& value) {
52 KVRequest request;

133TEST_F(KVExecutorTest, SetValue) {
134 std::map<std::string, std::string> data;
135
136 EXPECT_CALL(*mock_storage_ptr_, SetValue("test_key", "test_value"))
137 .WillOnce(Invoke([&](const std::string& key, const std::string& value) {
138 data[key] = value;
139 return 0;
140 }));
141
142 EXPECT_CALL(*mock_storage_ptr_, GetValue("test_key"))
143 .WillOnce(Invoke([&](const std::string& key) {
144 std::string ret = data[key];
145 return ret;
146 }));
147
148 EXPECT_CALL(*mock_storage_ptr_, GetAllValues())
149 .WillOnce(Return("[]"))
150 .WillOnce(Return("[test_value]"));
151
152 EXPECT_CALL(*mock_storage_ptr_, GetRange("a", "z"))
153 .WillOnce(Return("[test_value]"));
154
155 EXPECT_EQ(GetAllValues(), "[]");
156 EXPECT_EQ(Set("test_key", "test_value"), 0);
157 EXPECT_EQ(Get("test_key"), "test_value");
158
159 // GetAllValues and GetRange may be out of order for in-memory, so we test up to
160 // 1 key-value pair
161 EXPECT_EQ(GetAllValues(), "[test_value]");
162 EXPECT_EQ(GetRange("a", "z"), "[test_value]");
163}
