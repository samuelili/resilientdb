46 void Push(T&& data) {
47 std::lock_guard<std::mutex> lk(mutex_);
48 if (queue_.empty() || queue_.back()->list.size() >= batch_size_) {
49 queue_.push_back(std::make_unique<BatchQueueItem>());
50 queue_.back()->list.reserve(batch_size_);
51 }
52 queue_.back()->list.push_back(std::move(data));
53 cv_.notify_all();
54 }
55
56 size_t Size() { return queue_.size(); }
57
58 std::vector<T> Pop(int timeout_ms) {
59 std::unique_ptr<BatchQueueItem> item = nullptr;
60 {
61 std::unique_lock<std::mutex> lk(mutex_);
62 if (timeout_ms > 0) {
63 cv_.wait_for(lk, std::chrono::microseconds(timeout_ms), [&] {
64 return !queue_.empty() && queue_.front()->list.size() >= batch_size_;
65 });
66 }
67 if (queue_.empty()) {
68 return std::vector<T>();
69 }
70 item = std::move(queue_.front());
71 queue_.pop_front();
72 }
73 return std::move(item->list);
74 }
