65void ServiceNetwork::AcceptorHandler(const char* buffer, size_t data_len) {
66 BroadcastData data;
67 if (!data.ParseFromArray(buffer, data_len)) {
68 LOG(ERROR) << "parse broad cast fail:" << data_len;
69 return;
70 }
71
72 for (auto& sub_data : data.data()) {
73 std::unique_ptr<DataInfo> sub_request_info = std::make_unique<DataInfo>();
74 sub_request_info->data_len = sub_data.size();
75 sub_request_info->buff = new char[sub_request_info->data_len];
76 memcpy(sub_request_info->buff, sub_data.data(), sub_request_info->data_len);
77 std::unique_ptr<QueueItem> item = std::make_unique<QueueItem>();
78 item->socket = nullptr;
79 item->data = std::move(sub_request_info);
80 // LOG(ERROR) << "receve data from acceptor:" << data.is_resp()<<" data
81 // len:"<<item->data->data_len;
82 global_stats_->ServerCall();
83 input_queue_.Push(std::move(item));
84 }
85}

87void ServiceNetwork::InputProcess() {
88 std::vector<std::thread> threads;
89
90 int woker_num = config_.GetWorkerNum();
91 LOG(ERROR) << "server:" << config_.GetSelfInfo().id() << " start running";
92 for (int i = 0; i < woker_num; ++i) {
93 threads.push_back(std::thread([&]() {
94 while (IsRunning()) {
95 std::unique_ptr<QueueItem> item = input_queue_.Pop(1000);
96 if (item == nullptr) {
97 continue;
98 }
99 global_stats_->ServerProcess();
100 Process(std::move(item));
101 }
102 }));
103 }
104
105 for (auto& th : threads) {
106 if (th.joinable()) {
107 th.join();
108 }
109 }
110}

121// Receive a message from network and pass it to service to process.
122void ServiceNetwork::Process(std::unique_ptr<QueueItem> item) {
123 auto client_socket =
124 item->socket == nullptr ? nullptr : std::move(item->socket);
125 auto request_info = std::move(item->data);
126 if (client_socket != nullptr) {
127 client_socket->SetSendTimeout(1000000);
128 }
129 std::unique_ptr<Context> context = std::make_unique<Context>();
130 context->client = std::make_unique<NetChannel>(std::move(client_socket),
131 /*connected=*/true);
132 if (request_info) {
133 service_->Process(std::move(context), std::move(request_info));
134 }
135 return;
136}
