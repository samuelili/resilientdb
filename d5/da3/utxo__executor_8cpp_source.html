41std::unique_ptr<std::string> UTXOExecutor::ExecuteData(
42 const std::string& client_request) {
43 UTXORequest utxo_request;
44 UTXOResponse response;
45
46 if (!utxo_request.ParseFromString(client_request)) {
47 LOG(ERROR) << "parse data fail";
48 return nullptr;
49 }
50
51 int64_t ret = transaction_->AddTransaction(utxo_request.utxo());
52 response.set_ret(ret);
53 std::unique_ptr<std::string> ret_str = std::make_unique<std::string>();
54 response.SerializeToString(ret_str.get());
55 return ret_str;
56}

61std::unique_ptr<std::string> QueryExecutor::Query(
62 const std::string& request_str) {
63 UTXOQuery query;
64 if (!query.ParseFromString(request_str)) {
65 LOG(ERROR) << "parse query fail";
66 return nullptr;
67 }
68
69 UTXOQueryResponse resp;
70 if (query.query_transaction()) {
71 std::vector<UTXO> utxos =
72 transaction_->GetUTXO(query.end_id(), query.num());
73 for (const UTXO& utxo : utxos) {
74 *resp.add_utxos() = utxo;
75 }
76 } else {
77 resp.set_value(wallet_->GetCoin(query.address()));
78 }
79 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
80 resp.SerializeToString(resp_str.get());
81 return resp_str;
82}
