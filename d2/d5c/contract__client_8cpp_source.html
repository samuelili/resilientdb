52absl::StatusOr<Contract> ContractClient::DeployContract(
53 const std::string& caller_address, const std::string& contract_name,
54 const std::string& contract_path,
55 const std::vector<std::string>& init_params) {
56 std::ifstream contract_fstream(contract_path);
57 if (!contract_fstream) {
58 LOG(ERROR) << "could not find contract definition. file:" << contract_path
59 << " name:" << contract_name;
60 return absl::InvalidArgumentError("Contract not exist.");
61 }
62
63 nlohmann::json contracts_definition = nlohmann::json::parse(contract_fstream);
64
65 const auto all_contracts = contracts_definition["contracts"];
66
67 const std::string contract_code = all_contracts[contract_name]["bin"];
68 if (contract_code.empty()) {
69 LOG(ERROR) << "could not find contract definition. file:" << contract_path
70 << " name:" << contract_name;
71 return absl::InvalidArgumentError("Contract not exist.");
72 }
73
74 const auto func_hashes = all_contracts[contract_name]["hashes"];
75
76 DeployInfo deploy_info;
77 deploy_info.set_contract_bin(contract_code);
78 deploy_info.set_contract_name(contract_name);
79
80 for (auto& func : func_hashes.items()) {
81 FuncInfo* new_func = deploy_info.add_func_info();
82 new_func->set_func_name(func.key());
83 new_func->set_hash(func.value());
84 }
85
86 for (const std::string& param : init_params) {
87 deploy_info.add_init_param(param);
88 }
89
90 Request request;
91 Response response;
92 request.set_caller_address(caller_address);
93 *request.mutable_deploy_info() = deploy_info;
94
95 request.set_cmd(Request::DEPLOY);
96 LOG(ERROR) << "send request:" << request.DebugString();
97 int ret = SendRequest(request, &response);
98 if (ret != 0 || response.ret() != 0) {
99 return absl::InternalError("Deploy contract fail.");
100 }
101 return response.contract();
102}

104absl::StatusOr<std::string> ContractClient::ExecuteContract(
105 const std::string& caller_address, const std::string& contract_address,
106 const std::string& func_name, const std::vector<std::string>& func_params) {
107 Request request;
108 Response response;
109 request.set_caller_address(caller_address);
110 request.set_contract_address(contract_address);
111
112 request.mutable_func_params()->set_func_name(func_name);
113 for (const std::string& param : func_params) {
114 request.mutable_func_params()->add_param(param);
115 }
116
117 request.set_cmd(Request::EXECUTE);
118 LOG(ERROR) << "send request:" << request.DebugString();
119 int ret = SendRequest(request, &response);
120 if (ret != 0 || response.ret() != 0) {
121 return absl::InternalError("Deploy contract fail.");
122 }
123 return response.res();
124}
