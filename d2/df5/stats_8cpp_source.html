82void Stats::MonitorGlobal() {
83 LOG(ERROR) << "monitor:" << name_ << " sleep time:" << monitor_sleep_time_;
84
85 uint64_t seq_fail = 0;
86 uint64_t client_call = 0, socket_recv = 0;
87 uint64_t num_client_req = 0, num_propose = 0, num_prepare = 0, num_commit = 0,
88 pending_execute = 0, execute = 0, execute_done = 0;
89 uint64_t broad_cast_msg = 0, send_broad_cast_msg = 0;
90 uint64_t send_broad_cast_msg_per_rep = 0;
91 uint64_t server_call = 0, server_process = 0;
92 uint64_t seq_gap = 0;
93 uint64_t total_request = 0, total_geo_request = 0, geo_request = 0;
94
95 // ====== for client proxy ======
96 uint64_t run_req_num = 0, run_req_run_time = 0;
97
98 uint64_t last_run_req_num = 0, last_run_req_run_time = 0;
99 // =============================
100
101 uint64_t last_seq_fail = 0;
102 uint64_t last_num_client_req = 0, last_num_propose = 0, last_num_prepare = 0,
103 last_num_commit = 0;
104 uint64_t last_pending_execute = 0, last_execute = 0, last_execute_done = 0;
105 uint64_t last_client_call = 0, last_socket_recv = 0;
106 uint64_t last_broad_cast_msg = 0, last_send_broad_cast_msg = 0;
107 uint64_t last_send_broad_cast_msg_per_rep = 0;
108 uint64_t last_server_call = 0, last_server_process = 0;
109 uint64_t last_total_request = 0, last_total_geo_request = 0,
110 last_geo_request = 0;
111 uint64_t time = 0;
112
113 while (!stop_) {
114 sleep(monitor_sleep_time_);
115 time += monitor_sleep_time_;
116 seq_fail = seq_fail_;
117 socket_recv = socket_recv_;
118 client_call = client_call_;
119 num_client_req = num_client_req_;
120 num_propose = num_propose_;
121 num_prepare = num_prepare_;
122 num_commit = num_commit_;
123 pending_execute = pending_execute_;
124 execute = execute_;
125 execute_done = execute_done_;
126 broad_cast_msg = broad_cast_msg_;
127 send_broad_cast_msg = send_broad_cast_msg_;
128 send_broad_cast_msg_per_rep = send_broad_cast_msg_per_rep_;
129 server_call = server_call_;
130 server_process = server_process_;
131 seq_gap = seq_gap_;
132 total_request = total_request_;
133 total_geo_request = total_geo_request_;
134 geo_request = geo_request_;
135
136 run_req_num = run_req_num_;
137 run_req_run_time = run_req_run_time_;
138
139 LOG(ERROR) << "=========== monitor =========\n"
140 << "server call:" << server_call - last_server_call
141 << " server process:" << server_process - last_server_process
142 << " socket recv:" << socket_recv - last_socket_recv
143 << " "
144 "client call:"
145 << client_call - last_client_call
146 << " "
147 "client req:"
148 << num_client_req - last_num_client_req
149 << " "
150 "broad_cast:"
151 << broad_cast_msg - last_broad_cast_msg
152 << " "
153 "send broad_cast:"
154 << send_broad_cast_msg - last_send_broad_cast_msg
155 << " "
156 "per send broad_cast:"
157 << send_broad_cast_msg_per_rep - last_send_broad_cast_msg_per_rep
158 << " "
159 "propose:"
160 << num_propose - last_num_propose
161 << " "
162 "prepare:"
163 << (num_prepare - last_num_prepare)
164 << " "
165 "commit:"
166 << (num_commit - last_num_commit)
167 << " "
168 "pending execute:"
169 << pending_execute - last_pending_execute
170 << " "
171 "execute:"
172 << execute - last_execute
173 << " "
174 "execute done:"
175 << execute_done - last_execute_done << " seq gap:" << seq_gap
176 << " total request:" << total_request - last_total_request
177 << " txn:" << (total_request - last_total_request) / 5
178 << " total geo request:"
179 << total_geo_request - last_total_geo_request
180 << " total geo request per:"
181 << (total_geo_request - last_total_geo_request) / 5
182 << " geo request:" << (geo_request - last_geo_request)
183 << " "
184 "seq fail:"
185 << seq_fail - last_seq_fail << " time:" << time
186 << " "
187 "\n--------------- monitor ------------";
188 if (run_req_num - last_run_req_num > 0) {
189 LOG(ERROR) << " req client latency:"
190 << static_cast<double>(run_req_run_time -
191 last_run_req_run_time) /
192 (run_req_num - last_run_req_num) / 1000000000.0;
193 }
194
195 last_seq_fail = seq_fail;
196 last_socket_recv = socket_recv;
197 last_client_call = client_call;
198 last_num_client_req = num_client_req;
199 last_num_propose = num_propose;
200 last_num_prepare = num_prepare;
201 last_num_commit = num_commit;
202 last_pending_execute = pending_execute;
203 last_execute = execute;
204 last_execute_done = execute_done;
205
206 last_broad_cast_msg = broad_cast_msg;
207 last_send_broad_cast_msg = send_broad_cast_msg;
208 last_send_broad_cast_msg_per_rep = send_broad_cast_msg_per_rep;
209
210 last_server_call = server_call;
211 last_server_process = server_process;
212
213 last_run_req_num = run_req_num;
214 last_run_req_run_time = run_req_run_time;
215 last_total_request = total_request;
216 last_total_geo_request = total_geo_request;
217 last_geo_request = geo_request;
218 }
219}
