36void WriteKey(const std::string& key, const std::string& file_name) {
37 printf("save key to path %s\n", file_name.c_str());
38 int fd = open(file_name.c_str(), O_WRONLY | O_CREAT | O_TRUNC, 0666);
39 if (fd < 0) {
40 printf("open file %s fail %s\n", file_name.c_str(), strerror(errno));
41 return;
42 }
43 assert(fd >= 0);
44 write(fd, key.c_str(), key.size());
45 close(fd);
46}

48KeyInfo ReadKey(const std::string& file_name) {
49 int fd = open(file_name.c_str(), O_RDONLY, 0666);
50 if (fd < 0) {
51 printf("open file %s fail %s\n", file_name.c_str(), strerror(errno));
52 }
53 assert(fd >= 0);
54
55 std::string res;
56 int read_len = 0;
57 char tmp[1024];
58 while (true) {
59 read_len = read(fd, tmp, sizeof(tmp));
60 if (read_len <= 0) {
61 break;
62 }
63 res.append(tmp, read_len);
64 }
65 close(fd);
66 KeyInfo key;
67 assert(key.ParseFromString(res));
68 return key;
69}

81int main(int argc, char** argv) {
82 std::string path = "./";
83 std::string admin_pri_key_path, admin_pub_key_path;
84 std::string node_pub_key_path;
85 std::string ip;
86 std::string type;
87 int port = 0;
88 int64_t node_id = 0;
89 if (argc < 9) {
90 printf(
91 "<save path> <administor private key path> <adminisotr public key "
92 "path> <node public key path> <node id> <ip> <port> "
93 "<type>(client/server)\n");
94 exit(0);
95 } else {
96 path = argv[1];
97 admin_pri_key_path = argv[2];
98 admin_pub_key_path = argv[3];
99 node_pub_key_path = argv[4];
100 node_id = strtoull(argv[5], NULL, 10);
101 ip = argv[6];
102 port = strtoull(argv[7], NULL, 10);
103 type = argv[8];
104 }
105 // Read the keys of current node.
106 KeyInfo pub_key = ReadKey(node_pub_key_path);
107
108 // Read the keys of administor.
109 KeyInfo admin_pri_key = ReadKey(admin_pri_key_path);
110 KeyInfo admin_pub_key = ReadKey(admin_pub_key_path);
111
112 CertificateInfo info;
113 info.mutable_admin_public_key()->set_key(admin_pub_key.key());
114 info.mutable_admin_public_key()->set_hash_type(admin_pub_key.hash_type());
115 *info.mutable_public_key()->mutable_public_key_info()->mutable_key() =
116 pub_key;
117 info.mutable_public_key()->mutable_public_key_info()->set_node_id(node_id);
118 if ("client" == type) {
119 info.mutable_public_key()->mutable_public_key_info()->set_type(
120 CertificateKeyInfo::CLIENT);
121 } else {
122 info.mutable_public_key()->mutable_public_key_info()->set_type(
123 CertificateKeyInfo::REPLICA);
124 }
125 info.mutable_public_key()->mutable_public_key_info()->set_ip(ip);
126 info.mutable_public_key()->mutable_public_key_info()->set_port(port);
127
128 info.set_node_id(node_id);
129
130 // Get certificate using the key from administor.
131 SignatureVerifier verifier(
132 admin_pri_key,
133 GenerateCertificateInfo(admin_pub_key, admin_pri_key, admin_pub_key, 0));
134 auto resp_info =
135 verifier.SignCertificateKeyInfo(info.public_key().public_key_info());
136 assert(resp_info.ok());
137 *info.mutable_public_key()->mutable_certificate() = *resp_info;
138 std::cout << "info:" << info.DebugString();
139 // Write the keys with certificate to the file.
140 std::string str;
141 assert(info.SerializeToString(&str));
142 WriteKey(str, path + "/cert_" + std::to_string(node_id) + ".cert");
143}
