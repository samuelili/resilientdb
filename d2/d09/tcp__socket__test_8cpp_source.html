36TEST(TcpSocket, SendAndRecv) {
37 std::promise<bool> svr_done, cli_done;
38 std::future<bool> svr_done_future = svr_done.get_future(),
39 cli_done_future = cli_done.get_future();
40
41 std::thread svr_thead = std::thread([&]() {
42 TcpSocket svr_socket;
43 svr_socket.Listen("127.0.0.1", 1234);
44 svr_done.set_value(true);
45 auto client_socket = svr_socket.Accept();
46 char *buf = nullptr;
47 size_t len = 0;
48 int ret = client_socket->Recv((void **)&buf, &len);
49 EXPECT_EQ(ret, 4);
50 EXPECT_EQ(std::string(buf, len), "test");
51 free(buf);
52
53 ret = client_socket->Send("recv_suc");
54 EXPECT_EQ(ret, 0);
55
56 cli_done_future.get();
57 });
58
59 svr_done_future.get();
60 // Sever is ready, start the client.
61
62 TcpSocket client_socket;
63 int ret = client_socket.Connect("127.0.0.1", 1234);
64 ASSERT_EQ(ret, 0);
65 ret = client_socket.Send("test");
66 ASSERT_EQ(ret, 0);
67
68 char *buf = nullptr;
69 size_t len = 0;
70 ret = client_socket.Recv((void **)&buf, &len);
71 EXPECT_EQ(ret, 8);
72 EXPECT_EQ(std::string(buf, len), "recv_suc");
73 free(buf);
74
75 cli_done.set_value(true);
76 svr_thead.join();
77}
