41 int AddRequest(LockFreeCollectorPool* pool, uint64_t seq) {
42 Request request;
43 request.set_seq(seq);
44
45 TransactionCollector* collector = pool->GetCollector(seq);
46 if (collector == nullptr) {
47 return 1;
48 }
49 return collector->AddRequest(
50 std::make_unique<Request>(request), SignatureInfo(), false,
51 [](const Request&, int, TransactionCollector::CollectorDataType*,
52 std::atomic<TransactionStatue>*, bool) {});
53 }
54};

56TEST_F(CollectorPoolTest, AddRequest) {
57 LockFreeCollectorPool pool("test", 2, nullptr);
58 for (int i = 0; i < 16; ++i) {
59 int ret = AddRequest(&pool, i);
60 EXPECT_EQ(ret, 0);
61 }
62 int ret = AddRequest(&pool, 16);
