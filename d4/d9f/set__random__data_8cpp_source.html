38std::string Exec(const char* cmd) {
39 std::array<char, 128> buffer;
40 std::string result;
41
42 auto pipe = popen(cmd, "r"); // get rid of shared_ptr
43
44 if (!pipe) throw std::runtime_error("popen() failed!");
45
46 while (!feof(pipe)) {
47 if (fgets(buffer.data(), 128, pipe) != nullptr) result += buffer.data();
48 }
49
50 auto rc = pclose(pipe);
51
52 if (rc == EXIT_SUCCESS) { // == 0
53
54 } else if (rc == EXIT_FAILURE) { // EXIT_FAILURE is not used by all programs,
55 // maybe needs some adaptation.
56 }
57 return result;
58}
