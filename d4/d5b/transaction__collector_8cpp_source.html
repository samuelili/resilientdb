75int TransactionCollector::AddRequest(
76 std::unique_ptr<Request> request, const SignatureInfo& signature,
77 bool is_main_request,
78 std::function<void(const Request&, int received_count, CollectorDataType*,
79 std::atomic<TransactionStatue>* status, bool force)>
80 call_back) {
81 if (request == nullptr) {
82 LOG(ERROR) << "request empty";
83 return -2;
84 }
85
86 int32_t sender_id = request->sender_id();
87 std::string hash = request->hash();
88 int type = request->type();
89 uint64_t seq = request->seq();
90 uint64_t view = request->current_view();
91 if (is_committed_) {
92 return -2;
93 }
94 if (status_.load() == EXECUTED) {
95 return -2;
96 }
97
98 if (seq_ != static_cast<uint64_t>(request->seq())) {
99 // LOG(ERROR) << "data invalid, seq not the same:" << seq
100 // << " collect seq:" << seq_;
101 return -2;
102 }
103
104 if (is_main_request) {
105 auto request_info = std::make_unique<RequestInfo>();
106 request_info->signature = signature;
107 request_info->request = std::move(request);
108 bool force = false;
109 if (view_ && view_ < view && !is_prepared_) {
110 force = true;
111 atomic_mian_request_.Clear();
112 }
113 int ret = atomic_mian_request_.Set(request_info);
114 if (!ret) {
115 other_main_request_.insert(std::move(request_info));
116 LOG(ERROR) << "set main request fail: data existed:" << seq
117 << " ret:" << ret;
118 return -2;
119 }
120 auto main_request = atomic_mian_request_.Reference();
121 if (main_request->request == nullptr) {
122 LOG(ERROR) << "set main request data fail";
123 return -2;
124 }
125 view_ = view;
126 call_back(*main_request->request.get(), 1, nullptr, &status_, force);
127 return 0;
128 } else {
129 if (enable_viewchange_) {
130 if (type == Request::TYPE_PREPARE) {
131 if (status_.load() <= TransactionStatue::READY_PREPARE) {
132 auto request_info = std::make_unique<RequestInfo>();
133 request_info->signature = signature;
134 request_info->request = std::make_unique<Request>(*request);
135 std::lock_guard<std::mutex> lk(mutex_);
136 if (is_prepared_) {
137 return 0;
138 }
139 prepared_proof_.push_back(std::move(request_info));
140 if (senders_[type].count(hash) == 0) {
141 senders_[type].insert(std::make_pair(hash, std::bitset<128>()));
142 }
143 senders_[type][hash][sender_id] = 1;
144 call_back(*request, senders_[type][hash].count(), nullptr, &status_,
145 false);
146 if (status_.load() == TransactionStatue::READY_COMMIT) {
147 is_prepared_ = true;
148 if (atomic_mian_request_.Reference() != nullptr &&
149 atomic_mian_request_.Reference()->request->hash() != hash) {
150 atomic_mian_request_.Clear();
151 for (auto it = other_main_request_.begin();
152 it != other_main_request_.end(); it++) {
153 if ((*it)->request->hash() == hash) {
154 auto request_info = std::make_unique<RequestInfo>();
155 request_info->signature = (*it)->signature;
156 request_info->request = std::move((*it)->request);
157 atomic_mian_request_.Set(request_info);
158 break;
159 }
160 }
161 other_main_request_.clear();
162 }
163 int pos = 0;
164 for (size_t i = 0; i < prepared_proof_.size(); i++) {
165 if (prepared_proof_[i]->request->hash() == hash) {
166 prepared_proof_[pos++] = std::move(prepared_proof_[i]);
167 }
168 }
169 prepared_proof_.erase(prepared_proof_.begin() + pos,
170 prepared_proof_.end());
171 }
172 }
173 return 0;
174 }
175 }
176 if (request->type() == Request::TYPE_COMMIT) {
177 if (request->has_data_signature() &&
178 request->data_signature().node_id() > 0) {
179 std::lock_guard<std::mutex> lk(mutex_);
180 LOG(ERROR) << "add qc signature";
181 commit_certs_.push_back(request->data_signature());
182 }
183 }
184
185 {
186 std::lock_guard<std::mutex> lk(mutex_);
187 if (senders_[type].count(hash) == 0) {
188 senders_[type].insert(std::make_pair(hash, std::bitset<128>()));
189 }
190 senders_[type][hash][sender_id] = 1;
191 call_back(*request, senders_[type][hash].count(), nullptr, &status_,
192 false);
193 }
194
195 if (status_.load() == TransactionStatue::READY_EXECUTE) {
196 Commit();
197 return 1;
198 }
199 }
200 return 0;
201}

203int TransactionCollector::Commit() {
204 TransactionStatue old_status = TransactionStatue::READY_EXECUTE;
205 bool res = status_.compare_exchange_strong(
206 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
207 std::memory_order_acq_rel);
208 if (!res) {
209 return -2;
210 }
211
212 auto main_request = atomic_mian_request_.Reference();
213 if (main_request == nullptr) {
214 LOG(ERROR) << "no main";
215 return -2;
216 }
217
218 is_committed_ = true;
219 if (executor_ && main_request->request) {
220 if (!commit_certs_.empty()) {
221 for (const auto& sig : commit_certs_) {
222 *main_request->request->mutable_committed_certs()
223 ->add_committed_certs() = sig;
224 // LOG(ERROR) << "add sig:" << sig.DebugString();
225 }
226 }
227 executor_->Commit(std::move(main_request->request));
228 }
229 return 0;
230}
