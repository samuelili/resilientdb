47std::string ChainState::GetValue(const std::string& key) {
48 if (storage_) {
49 return storage_->GetValue(key);
50 }
51 auto search = kv_map_.find(key);

59std::string ChainState::GetAllValues(void) {
60 if (storage_) {
61 return storage_->GetAllValues();
62 }
63 std::string values = "[";
64 bool first_iteration = true;
65 for (auto kv : kv_map_) {
66 if (!first_iteration) values.append(",");
67 first_iteration = false;
68 values.append(kv.second);
69 }
70 values.append("]");
71 return values;

74std::string ChainState::GetRange(const std::string& min_key,
75 const std::string& max_key) {
76 if (storage_) {
77 return storage_->GetRange(min_key, max_key);
78 }
79 std::string values = "[";
80 bool first_iteration = true;
81 for (auto kv : kv_map_) {
82 if (kv.first >= min_key && kv.first <= max_key) {
83 if (!first_iteration) values.append(",");
84 first_iteration = false;
85 values.append(kv.second);
86 }
87 }
88 values.append("]");
89 return values;
