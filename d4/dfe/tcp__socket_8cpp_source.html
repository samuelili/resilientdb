41int RecvInternal(int fd, void* buf, size_t len) {
42 size_t pos = 0;
43 while (len > 0) {
44 int ret = recv(fd, (char*)buf + pos, len, 0);
45 if (ret <= 0) {
46 // LOG(ERROR) << "recv data fail, fd =" << fd << " error " <<
47 // strerror(errno)
48 // << " ret:" << ret;
49 return ret;
50 }
51 pos += ret;
52 len -= ret;
53 }
54 return pos;
55}

57int SendInternal(int fd, const void* buf, size_t len) {
58 if (fd < 0) {
59 return -2;
60 }
61 size_t pos = 0;
62 while (pos < len) {
63 int ret = send(fd, (char*)buf + pos, len - pos, 0);
64 if (ret < 0) {
65 LOG(ERROR) << "send data fail, fd =" << fd << " error "
66 << strerror(errno);
67 return -1;
68 }
69 pos += ret;
70 }
71 return pos;
72}

103int TcpSocket::Listen(const std::string& ip, int port) {
104 int on = 1;
105 if (setsockopt(socket_fd_, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) != 0) {
106 LOG(ERROR) << "set TcpSocket opt fail, error" << strerror(errno);
107 return -1;
108 }
109
110 struct sockaddr_in servaddr;
111 memset(&servaddr, 0, sizeof(servaddr));
112 servaddr.sin_family = AF_INET;
113 servaddr.sin_addr.s_addr = inet_addr(ip.data());
114 servaddr.sin_port = htons(port);
115
116 if (bind(socket_fd_, (struct sockaddr*)&servaddr, sizeof(servaddr)) == -1) {
117 LOG(ERROR) << "bind TcpSocket error: " << strerror(errno)
118 << "(errno: " << errno << ")";
119 return -1;
120 }
121
122 if (listen(socket_fd_, 10) == -1) {
123 LOG(ERROR) << "listen TcpSocket error: " << strerror(errno)
124 << "(errno: " << errno << ")";
125 return -1;
126 }
127 if (port == 0) {
128 struct sockaddr_in localaddr;
129 socklen_t len = sizeof(localaddr);
130 int ret = getsockname(socket_fd_, (struct sockaddr*)&localaddr, &len);
131 if (ret != 0) {
132 LOG(ERROR) << "get binding port fail:" << strerror(errno);
133 } else {
134 binding_port_ = ntohs(localaddr.sin_port);
135 }
136 } else {
137 binding_port_ = port;
138 }
139 return 0;
140}

183int TcpSocket::Send(const std::string& data) {
184 size_t data_size = data.size();
185 size_t send_ret = SendInternal(socket_fd_, &data_size, sizeof(data_size));
186 if (send_ret != sizeof(data_size)) {
187 if (static_cast<int>(send_ret) == -1) {
188 LOG(ERROR) << "send data error: " << strerror(errno)
189 << "(errno: " << errno << ")";
190 return -1;
191 }
192 return -2;
193 }
194
195 send_ret = SendInternal(socket_fd_, data.c_str(), data_size);
196 if (send_ret != data_size) {
197 if (static_cast<int>(send_ret) == -1) {
198 LOG(ERROR) << "send data not done: " << strerror(errno)
199 << "(errno: " << errno << ")";
200 return -1;
201 } else {
202 return -2;
203 }
204 }
205 return 0;
206}

208int TcpSocket::Recv(void** buf, size_t* len) {
209 int ret = RecvInternal(socket_fd_, len, sizeof(size_t));
210 if (ret <= 0) {
211 return ret;
212 }
213 *buf = malloc(*len);
214 ret = RecvInternal(socket_fd_, *buf, *len);
215 if (ret <= 0) {
216 return ret;
217 }
218 return ret;
