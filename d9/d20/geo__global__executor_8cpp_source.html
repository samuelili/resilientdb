53void GeoGlobalExecutor::Execute(std::unique_ptr<Request> request) {
54 BatchUserRequest batch_request;
55 if (!batch_request.ParseFromString(request->data())) {
56 LOG(ERROR) << "[GeoGlobalExecutor] parse data fail!";
57 return;
58 }
59 if (request->data().empty()) {
60 LOG(ERROR) << "[GeoGlobalExecutor] request->data() is empty ";
61 return;
62 }
63
64 if (global_stats_) {
65 global_stats_->IncTotalGeoRequest(batch_request.user_requests_size());
66 }
67 if (global_transaction_manager_) {
68 auto batch_response =
69 global_transaction_manager_->ExecuteBatch(batch_request);
70 if (batch_response != nullptr &&
71 request->region_info().region_id() == my_region_) {
72 batch_response->set_createtime(batch_request.createtime());
73 batch_response->set_local_id(batch_request.local_id());
74 batch_response->set_proxy_id(batch_request.proxy_id());
75 batch_response->set_seq(batch_request.seq());
76 resp_queue_.Push(std::move(batch_response));
77 }
78 }
79}

88void GeoGlobalExecutor::AddData() {
89 auto request = order_queue_.Pop();
90 if (request == nullptr) {
91 return;
92 }
93 global_stats_->IncGeoRequest();
94 uint64_t seq_num = request->seq();
95 int region_id = request->region_info().region_id();
96 execute_map_[std::make_pair(seq_num, region_id)] = std::move(request);
97}

99std::unique_ptr<Request> GeoGlobalExecutor::GetNextMap() {
100 if (execute_map_.empty()) {
101 return nullptr;
102 }
103 // LOG(ERROR) << "Next round seq: " << next_seq_ << ", current First in map: "
104 // << execute_map_.begin()->first << ", size: " <<
105
106 // LOG(ERROR)<<"get next seq:"<<next_seq_<<" region:"<<next_region_;
107 if (execute_map_.begin()->first == std::make_pair(next_seq_, next_region_)) {
108 std::unique_ptr<Request> res = std::move(execute_map_.begin()->second);
109 execute_map_.erase(execute_map_.begin());
110 next_region_++;
111 if (next_region_ > static_cast<int64_t>(region_size_)) {
112 next_region_ = 1;
113 next_seq_++;
114 }
115 return res;
116 }
117 return nullptr;
118}

120void GeoGlobalExecutor::OrderRound() {
121 while (!IsStop()) {
122 AddData();
123 while (!IsStop()) {
124 std::unique_ptr<Request> seq_map = GetNextMap();
125 if (seq_map == nullptr) {
126 break;
127 }
128 Execute(std::move(seq_map));
129 }
130 }
131}
