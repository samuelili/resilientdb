57int Query::ProcessQuery(std::unique_ptr<Context> context,
58 std::unique_ptr<Request> request) {
59 QueryRequest query;
60 if (!query.ParseFromString(request->data())) {
61 LOG(ERROR) << "parse data fail";
62 return -2;
63 }
64 // LOG(ERROR) << "request:" << query.DebugString();
65
66 QueryResponse response;
67 for (uint64_t i = query.min_seq(); i <= query.max_seq(); ++i) {
68 Request* ret_request = message_manager_->GetRequest(i);
69 if (ret_request == nullptr) {
70 break;
71 }
72 Request* txn = response.add_transactions();
73 txn->set_data(ret_request->data());
74 txn->set_hash(ret_request->hash());
75 txn->set_seq(ret_request->seq());
76 txn->set_proxy_id(ret_request->proxy_id());
77 }
78
79 if (context != nullptr && context->client != nullptr) {
80 // LOG(ERROR) << "send response:" << response.DebugString();
81 int ret = context->client->SendRawMessage(response);
82 if (ret) {
83 LOG(ERROR) << "send resp fail ret:" << ret;
84 }
85 }
86 return 0;
87}

89int Query::ProcessCustomQuery(std::unique_ptr<Context> context,
90 std::unique_ptr<Request> request) {
91 if (custom_query_executor_ == nullptr) {
92 LOG(ERROR) << "no custom executor";
93 return -1;
94 }
95
96 std::unique_ptr<std::string> resp_str =
97 custom_query_executor_->Query(request->data());
98
99 CustomQueryResponse response;
100 if (resp_str != nullptr) {
101 response.set_resp_str(*resp_str);
102 }
103
104 if (context != nullptr && context->client != nullptr) {
105 int ret = context->client->SendRawMessage(response);
106 if (ret) {
107 LOG(ERROR) << "send resp fail ret:" << ret;
108 }
109 }
110 return 0;
111}
